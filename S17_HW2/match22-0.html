void insertionSort(vector<TokenFreq> & tokFreqVector) { // sorts tokens in decending order

    int i, j;
    TokenFreq tmp;

    for (i = 1; i < tokFreqVector.size(); i++) { // loops through the vector of tokens
        j = i;

        while (j != i) { // sorts through tokens

            tmp = tokFreqVector[j];
            tokFreqVector[j] = tokFreqVector[j - 1];
            tokFreqVector[j - 1] = tmp;
