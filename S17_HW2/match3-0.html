#include <iostream>
#include <string>
#include <vector>
#include <fstream>

using namespace std;

struct TokenFreq{
	string value;
	int freq;
};

vector<TokenFreq> getTokenFreq(string inFile_name){
	vector<TokenFreq> token;
	
	int i =0; //converts all chars to lowercase
	char c= ' ';
	while (inFile_name[i]){
		c= inFile_name[i];
		inFile_name[i]= (tolower(c));
		i++;
	}
	
	ifstream inFile(inFile_name.c_str());
	string temp;
	while (inFile >> temp){
		int index=-1;
		for (int i=0; i < token.size();i++){
			if (token[i].value.compare(temp)==0){
				index = i;
				break;
			}
		}
	if (index == -1){
		TokenFreq t= {temp, 1}; //inputs strnig & freq
		token.push_back(t);
	}
	else{
		token[index].freq++;
	}
	}
	inFile.close(); //closes the file
	return token;
}

void selectionSort(vector<TokenFreq> & tokFreqVector){ //sort a vector in ascending order of token frequency
	TokenFreq temp;
	int min;
	for (int i=0; i<tokFreqVector.size();i++){
		min= i;
		for (int j= i; j< tokFreqVector.size(); j++){
			if(tokFreqVector[i].freq< tokFreqVector[min].freq)
		min= j;
	}
		if (min != i){
			temp= tokFreqVector[i];
			tokFreqVector[i]= tokFreqVector[min];
			tokFreqVector [min]= temp;
		}
	}
	for(int i=0; i<tokFreqVector.size()-1;i++){
		cout<< tokFreqVector[i].value<< ": "<< tokFreqVector[i].freq<<endl;
	}
}

void insertionSort(vector<TokenFreq> & tokFreqVector){ //sort the vector in descending order of token frequency
	int i,j;
	TokenFreq tmp;
	for (int i =1; i < tokFreqVector.size(); i++){
		tmp = tokFreqVector[i];
		for (j= i-1;(j>=0)&& (tokFreqVector[j].freq< tmp.freq); j--){
			tokFreqVector[j+1]= tokFreqVector[j];
		}
		tokFreqVector[j+1]= tmp;
		}
		for(int k=0; k<tokFreqVector.size()-1;k++){
		cout<< tokFreqVector[k].value<< ": "<< tokFreqVector[k].freq<<endl;
	}
}


void writeToFile(vector<TokenFreq> &tokFreqV, string outFileName ){
	ofstream outFile(outFileName.c_str()); //create out file stream
	
 	for(int i = 0; i <tokFreqV.size();i++){
		outFile << tokFreqV[i].value<< "-"<<tokFreqV[i].freq<<endl;
	}
	outFile.close(); //closes the file
}

int main(){
	string fileName, file;
	
	cout << "Please enter the file name: ";
	getline(cin, fileName);
	
	vector<TokenFreq> letters= getTokenFreq(fileName);
	
	for (int i = 0; i < letters.size(); i++){
		cout<< letters[i].value<< "-" << letters[i].freq<<endl;
	}
	cout<< "To print the output for Selection Sort, enter the name of the file: ";
	cin>> file;
	selectionSort(letters);
	writeToFile(letters,file);
	
	cout<< "To print the output for Insertion Sort, enter the name of the file: ";
	cin>> file;
	insertionSort(letters);
	
	writeToFile(letters, file);

}
