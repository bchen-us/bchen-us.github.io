#include <cstdlib>
#include <vector>
#include <string>
#include <iostream>
#include <fstream>
#include <cstring>

using namespace std;

struct tokenFreq{string Value; int freq;};

vector<tokenFreq> getTokenFreq(string inFileName){
  
    ifstream inFile;
    inFile.open(inFileName);
    
    if(inFile.fail()){
        cout << "File failed to open, program terminating...";
        exit (EXIT_FAILURE);
    } 
    
    vector<string> tokens;
    string temp;
    while(getline(inFile, temp, ' ')){
        
    tokens.push_back(temp);}
          
    
    vector<tokenFreq> frequency(2);
    
    for(int j = 0; j<tokens.size();j++){
        tokenFreq temp;

        for(int k = 0; k<frequency.size();k++){
            
            if(tokens[j]==frequency[k].Value)
            {frequency[k].freq++;}
            
            else{
            temp = {tokens[j], 1};
            
            }

        }
        frequency.push_back(temp);
     
    }
       
    
        inFile.close();
        return frequency;
    }

    
    
    
    
    
    


void selectionSort( vector<tokenFreq> & tokFreqVector ){
    for(int i=0;i<tokFreqVector.size()-1;i++){
        
        int posMin = i;
    
        for(int j = i+1;j<tokFreqVector.size();j++){
        if (tokFreqVector[j].freq < tokFreqVector[posMin].freq)
        {posMin=j;} 
        }
            if (posMin!=i){
                tokenFreq temp = tokFreqVector[i];
                tokFreqVector[i] = tokFreqVector[posMin];
                tokFreqVector[posMin] = temp;
            }
    
    }
};//ascendingfreqorder

void insertionSort(vector<tokenFreq> & tokFreqVector) {
    int j;
    tokenFreq temp;
for (int i = 0; i < tokFreqVector.size(); i++){

		j = i;
                
		while (j > 0 && tokFreqVector[j].freq > tokFreqVector[j-1].freq){
                    temp = tokFreqVector[j];
                    
                    tokFreqVector[j] = tokFreqVector[j-1];
                    tokFreqVector[j-1] = temp;
                    j--;
			  }
		}
    
    
};//descendingorder


void writeToFile( vector<tokenFreq> &tokFreqV, string outFileName) {
    ofstream vectorPrint;
    vectorPrint.open(outFileName);
    for(int i = 0; i<tokFreqV.size();i++){
    vectorPrint<<"{"<<tokFreqV[i].Value<<": "<<tokFreqV[i].freq<<"},  "<<endl;}
    vectorPrint.close();
    
}

int main(int argc, char** argv) {
    string fileName;
    cout<<"Type name of input file: "<<endl;    
    getline(cin, fileName);
    cout<<"Your file name was: " << fileName<<endl; 
    
    vector<tokenFreq> RefVec = getTokenFreq(fileName);
    selectionSort(RefVec);
    writeToFile(RefVec, "SelectionSort.txt");
    insertionSort(RefVec);
    writeToFile(RefVec, "InsertionSort.txt");
    
    
    
    return 0;
}
