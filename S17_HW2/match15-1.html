vector<TokenFreq> getTokenFreq(string file_name )
{
    fstream fin;
    string word;
	string token;
    vector<TokenFreq> freq;
    fin.open(file_name.c_str());
    if (!fin.good()){
      cout<< "Fail to open the file "<<file_name<<endl;
      exit(-1);
    }
    while(fin.good())
    {  
      getline(fin, word);
      istringstream ss(word);
      while(getline(ss, token, ' ')){
          transform(token.begin(), token.end(), token.begin(), ::tolower);
             TokenFreq temp;            
         for(int i = 0; i < freq.size(); )
          {
            if(token == freq[i].value)
            {
               freq[i].frequency++;
                             
            }   
            else if(token != freq[i].value)  
            {
              
               temp.value = token;
               temp.frequency=1;
                    
            }
           i++;
          }
         freq.push_back(temp);

       }
      freq.erase(freq.begin());
      
    }
     
     fin.close();
            
    return freq;


}




void selectionSort( vector<TokenFreq> &tokFreqVector ){
      
      int i, j, minIndex;
      TokenFreq tmp;    
      for (i = 0; i < tokFreqVector.size() - 1; i++) {
            minIndex = i;
           for (j = i + 1; j < tokFreqVector.size(); j++)
                 if (tokFreqVector[j].frequency < tokFreqVector[minIndex].frequency)
                        minIndex = j;
            if (minIndex != i) {
                  tmp = tokFreqVector[i];
                  tokFreqVector[i] = tokFreqVector[minIndex];
                  tokFreqVector[minIndex] = tmp;
            }
      }
    
    
    
    
}


void insertionSort( vector<TokenFreq> & tokFreqVector ) {
      int i, j; 
      TokenFreq tmp;
      for (i = 1; i < tokFreqVector.size(); i++) {
            j = i;
            while (j > 0 && tokFreqVector[j].frequency > tokFreqVector[j-1].frequency) {
                  tmp = tokFreqVector[j];
                  tokFreqVector[j] = tokFreqVector[j - 1];
                  tokFreqVector[j - 1] = tmp;
                  j--;
            }
      }
}

     for(int i=0; i< tokFreqV.size(); i++){
         fout <<"( " <<tokFreqV[i].value << ", "<<tokFreqV[i].frequency << " )"<<endl;
         
     }
    
   fout.close(); 
    
}
