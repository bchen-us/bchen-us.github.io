void selectionSort(vector<TokenFreq> something)
{
    int i;//initialize
    int j;

    int lowest;
    TokenFreq temporary_place;
    for (i=0; i<= something.size()-1; i++)
	{
		lowest = i;// have the lowest index be the index ,reset
		for(j=i+1; j < something.size();j++)
		{
		                 if (something[j].freq < something[lowest].freq){
						 
                        lowest = j;}
            if (lowest != i) {
                  temporary_place = something[i];
                  something[i] = something[lowest];
                  something[lowest] = temporary_place;
            }
      }
}
 cout <<"this is selection sort:\n";
 
 for (int i = 0;i<something.size();i++){

}


void insertionSort(vector<TokenFreq> something)
{

{

      int i, j;
	  TokenFreq temporary_place;
      for (i = 1; i < something.size(); i++) {
            j = i;
            while (j > 0 && something[j - 1].freq < something[j].freq) {
                  temporary_place = something[j];
                  something[j] = something[j - 1];
                  something[j - 1] = temporary_place;
                  j--;
            }
      }
}
 cout <<"this is insertion sort:\n";
 
 for (int i = 0;i<something.size();i++){
 	cout<< "frequency is "<< something[i].freq <<'\t'<< something[i].Value<<'\n';
 }
}
