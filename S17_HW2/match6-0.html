vector<TokenFreq> getTokenFreq( string  inFile_name)
{
    ifstream file;
    vector<TokenFreq> pairs;    
    TokenFreq newtoken;
    file.open (inFile_name.c_str(), std::ifstream::in);
    string word, mean;
    if(file.is_open())
    {
    while ( getline (file,mean) )
    {
    istringstream line(mean);
    while (line >> word)
    {
    int count=0;
    for (int i=0; i< pairs.size(); i++)
    {
    if (iequals(pairs[i].value , word) )
    {
    pairs[i].frequancy++;    
    count=1;
    }
    }
    if (count==0)
    {
    newtoken.value = word;
    newtoken.frequancy=1;
    pairs.push_back(newtoken);
    }
    }
    }   
 }
    file.close();
    return pairs;
}
void selectionSort( vector<TokenFreq> & tokFreqVector )
{
    int i, j, k;
    TokenFreq temp;
    int tokenLength = tokFreqVector.size();
    for (i= tokenLength-1; 0<i; i--)
    {k= 0;                 
     for (j=1; j<=i; j++) 
    {
    if (tokFreqVector[j].frequancy < tokFreqVector[k].frequancy)
    k = j;
    }
    temp = tokFreqVector[k];   
    tokFreqVector[k] = tokFreqVector[i];
    tokFreqVector[i] = temp;
    }
     
}
bool iequals(const string& alpha, const string& betta)
{
    unsigned int supper = alpha.size();
    if (betta.size()!= supper)
    return false;
    for (unsigned int i = 0; i < supper; ++i)
    if (tolower(alpha[i]) != tolower(betta[i]))
	return false;
    return true;
}
