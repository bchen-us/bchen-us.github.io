using namespace std;

struct TokenFreq{
    string value; //token
    int freq = 0; //frequency
    
    
};


void getTokenFreq( string  inFile_name){
    ifstream aFile;
    string tmp;
    string text;
    inFile_name = "SteveJobsSpeech2005";
    
    
    aFile.open(inFile_name,ios::in|ios::out); //opens file
    if(aFile.is_open() && !aFile.eof()){
        while(std::getline(aFile,tmp)){ //gets each line of text and converts to a tmp string.
            vector<TokenFreq> tokens;   
            transform(tmp.begin(),tmp.end(),tmp.begin(),::tolower); //changes the tmp string to all lowercase.
            text.append(tmp);//copies the lines of strings and makes it all one long string
            char* token;
            token=strtok(&text[0], " ");
            
            while(token != NULL){
                TokenFreq word;
                word.value=token;
                word.freq=1;
                tokens.push_back(word);
                token = strtok(NULL, " ");
            }
            for(int i=0; i < tokens.size(); i++){
                for(int j=1; j < tokens.size(); j++){
                    if(TokenFreq.value[i]==TokenFreq.value[j]){
                        TokenFreq count;
                        count.freq+1;
                        
                    }
                    else{
                        TokenFreq word;
                        word.value=TokenFreq.value[i];
                        word.freq=1;
                        tokens.push_back(word);
                    }
                }
            }
        }
       
    }
    else{ //run if file doesn't open
        cout << "Unable to open file." << endl;
    }
    aFile.close(); //closes the file
}

int main(int arc, char* argv[]) {  
    
    vector <TokenFreq> tokens = getTokenFreq("SteveJobsSpeech2005"); //calls the function and vector.   
    writeToFile(tokens, "default_tokens");
    
    selectionSort(tokens); 
    writeToFile(tokens, "AscendingSelectionSort");
    
    insertionSort (tokens);
    writeToFile(tokens, "DescendingInsertionSort");

    return 0;
}

void selectionSort( vector<TokenFreq>&tokFreqVector ){
    TokenFreq tmp;
    
    for(int i=0; i < tokFreqVector.size(); i++){
        for(int j=1; j < tokFreqVector.size(); j++){    //compares values in a cell and the one after.
            if(tokFreqVector[j].freq <= tokFreqVector[i].freq){//if the value in the cell ahead is lower.
                tmp = tokFreqVector[i];                         //store the higher value in a temp
                tokFreqVector[i] = tokFreqVector[j];          //higher value shifts up one cell
                tmp = tokFreqVector[j];                         //temp value is reinstated to the cell closer to the start.
            }
        }
    }
}
    
void insertionSort( vector<TokenFreq>&tokFreqVector ){
    TokenFreq tmp;
    for (int i = 0; i < tokFreqVector.size(); i++) {
        for(int j=1; j < tokFreqVector.size(); j++){    //compares values in a cell and the one after.
            if(tokFreqVector[j].freq >= tokFreqVector[i].freq){//if the value in the cell ahead is higher.
                tmp = tokFreqVector[j];                         //store the lower value in a temp
                tokFreqVector[j] = tokFreqVector[i];          //lower value shifts up one cell
                tmp = tokFreqVector[i];                         //temp value is reinstated to the cell closer to the start.
            }
        }
    }
}

            outFile << tokFreqV;//place tokens in the file

        }

}
