#include "Polynomial.h"


polynomial::polynomial()
{
    degreeMax = 0;

    Coefficients = new double[degreeMax + 1];
    for(int i = 0; i < degreeMax + 1; ++i)
    {
        Coefficients[i] = 0;
    }
}

polynomial::polynomial(int highestDegree) //Figures out whats the highest degree
{
    degreeMax = highestDegree;

    Coefficients = new double[degreeMax + 1];
    for(int i = 0; i < degreeMax + 1; ++i)
    {
        Coefficients[i] = 0;
    }
}
polynomial polynomial::operator - () //Function for operator -
{
    polynomial p(degreeMax);

    for(int i = 0; i < degreeMax + 1; ++i)
    {
        p.setCoeff(i, -1 * getCoeff(i));
        setCoeff(i, -1 * getCoeff(i));
    }
    return p; //returns subtraction value
}


int polynomial::degree()
{

    for(int i = degreeMax; i >= degreeMax; --i)
    {
        if(Coefficients[i] != 0)
        {
            return i;
        }
    }

    return -1; //negative number or not
}
void polynomial::PolySet() //Setter function for input
{

    cout << "Please enter in maximum degree: ";
    cin >> degreeMax;
    if(Coefficients != NULL)
    {
        delete[] Coefficients;
    }
    Coefficients = new double[degreeMax + 1];
    for(int i = 0; i < degreeMax + 1; ++i)
    {
        cout << "Please enter in degree of coefficient " << i << ": ";
        cin >> Coefficients[i];

        double polynomial::getCoeff(int power)
        {
            if(power < 0 || power > degreeMax + 1)
                return 0;
            return Coefficients[power];
        }



    }
}

void polynomial::setCoeff(int power, double newCoefficient) //planting values into variables
{
    Coefficients[power] = newCoefficient;
}

polynomial polynomial::operator / (int val)
{
    polynomial craveHolder(degreeMax);

    for(int i = 0; i < degreeMax + 1; ++i)
    {
        craveHolder.setCoeff(i, getCoeff(i) / val);
    }
    return craveHolder;
}



polynomial polynomial::operator + (polynomial p) //Operator + function
{
    polynomial ret(degreeMax > p.degreeMax ? degreeMax : p.degreeMax);
    for(int i = 0; i < degreeMax + 1; ++i)
    {
        ret.setCoeff(i, getCoeff(i) + p.getCoeff(i));
    }
    return ret;
}


ostream& operator << (ostream &out, polynomial p)
{
    double holder;
    for(int i = p.degreeMax; i >= 0; --i)
    {
        holder = p.getCoeff(i);
        if(holder != 0)
        {
            if(holder > 0)
            {
                out << " +" << holder << "*x^" << i;
            }
            else
            {
                out << " " << holder << " * x^" << i;
            }
        }
    }
    return out;
}
polynomial polynomial::operator - (polynomial p)
{
    polynomial ret(degreeMax > p.degreeMax ? degreeMax : p.degreeMax);
    for(int i = 0; i < degreeMax + 1; ++i)
    {
        ret.setCoeff(i, getCoeff(i) - p.getCoeff(i));
    }
    return ret;
}
