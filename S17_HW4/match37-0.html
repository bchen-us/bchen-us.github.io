#include <iostream>
#include "Matrix.h"

Matrix::Matrix() {
	//initialize matrix to 0.0
	initialize(10);
}
//initializes a matrix by using values stored in a vector of vectors
Matrix::SquareMatrix(vector < vector<double> > v2d ) {
	bool matrixsize = true;
   	int matrixsize = v2d.size();
   	for(int i = 0; i < matrixsize; i++){
       	if(v2d[i].size() != matrixsize){
        	matrixsize = false;
       	}
   	}
   	if(matrixsize){
       	x.clear();
       	for(int i = 0; i < v2d.size(); i++){
           	vector<double> vector1;
           	for(int j = 0; j < v2d[i].size(); j++){
               	vector1.push_back(v2d[i][j]);
           	}
           	x.push_back(vector1);
      	}
   	}
}

//returns value in matrix
double Matrix::getValue(int i, int j) {
	if(i < x.size() && j < x.size() && i >= 0 && j >= 0){
       	return x[i][j];
   	}
   	return 0; 
}

//sets the element to value
void Matrix::setValue(int i, int j, double value) {
	if(i < x.size() && j < x.size() && i >= 0 && j >= 0){
    	x[i][j] = value;
   	}
}

ostream& operator << (ostream &out, Matrix m) {
	for(int i = 0; i < m.x.size(); ++i){
       for(int j = 0; j < m.x[i].size(); ++j){
           out << m.x[i][j] << " ";
