#include "Polynomial.h"

polynomial::polynomial(){
   max_degree = 0;
   coeffs = new double[max_degree + 1];
   for(int i = 0; i < max_degree + 1; ++i){
       coeffs[i] = 0;
   }
}

polynomial::polynomial(int highest_degree){
   max_degree = highest_degree;
   coeffs = new double[max_degree + 1];
   for(int i = 0; i < max_degree + 1; ++i){
       coeffs[i] = 0;
   }
}


int polynomial::degree(){
   for(int i = max_degree; i >= max_degree; --i){
       if(coeffs[i] != 0){
           return i;
       }
   }
   return -1;
}

double polynomial::getCoeff(int power){
   if(power < 0 || power > max_degree + 1)
       return 0;
   return coeffs[power];
}

void polynomial::setPolynomial(){
   cout << "Enter max degree: ";
   cin >> max_degree;
   if(coeffs != NULL){
       delete[] coeffs;
   }
   coeffs = new double[max_degree + 1];
   for(int i = 0; i < max_degree + 1; ++i){
       cout << "Enter coefficient of degree " << i << ": ";
       cin >> coeffs[i];
   }
}

void polynomial::setCoeff(int power, double newCoefficient){
   coeffs[power] = newCoefficient;
}

polynomial polynomial::operator / (int val){
   polynomial p(max_degree);
   for(int i = 0; i < max_degree + 1; ++i){
       p.setCoeff(i, getCoeff(i) / val);
   }
   return p;
}

polynomial polynomial::operator - (){
   polynomial p(max_degree);
   for(int i = 0; i < max_degree + 1; ++i){
       p.setCoeff(i, -1 * getCoeff(i));
       setCoeff(i, -1 * getCoeff(i));
   }
   return p;
}

polynomial polynomial::operator + (polynomial p){
   polynomial ret(max_degree > p.max_degree ? max_degree : p.max_degree);
   for(int i = 0; i < max_degree + 1; ++i){
       ret.setCoeff(i, getCoeff(i) + p.getCoeff(i));
   }
   return ret;
}

polynomial polynomial::operator - (polynomial p){
   polynomial ret(max_degree > p.max_degree ? max_degree : p.max_degree);
   for(int i = 0; i < max_degree + 1; ++i){
       ret.setCoeff(i, getCoeff(i) - p.getCoeff(i));
   }
   return ret;
}

ostream& operator << (ostream &out, polynomial p){
   double c;
   for(int i = p.max_degree; i >= 0; --i){
       c = p.getCoeff(i);
       if(c != 0){
           if(c > 0){
               out << " +" << c << "*x^" << i;
           }
           else{
               out << " " << c << " * x^" << i;
           }
       }
   }
   return out;
}
