#include "polynomial.h"
#include <iostream>

using namespace std;

polynomial::polynomial(){
    maxDegree = 0;
    coefficient = new double[maxDegree +1];
    
    for(int i = 0;i < maxDegree; i++){
        coefficient[i] = 0;
    }
}

polynomial::polynomial(int high_degree){
    maxDegree = high_degree;
    coefficient = new double[maxDegree + 1];
    for (int i = 0; i < maxDegree; i++) {
        coefficient[i] = 0;
    }
}

int polynomial::degree(){
    for (int i = maxDegree; i >= maxDegree; --i) {
        if (coefficient[i] != 0) {
            return i;
        }
    }
    return -1;
}

double polynomial::getCoEff(int power){
    if (power < 0 || power > maxDegree + 1) {
        return 0;
        return coefficient[power];
    }
    
    void polynomial::setPoly(){
        cout << "Enter Highest Degree: ";
        cin >> maxDegree;
        if (coefficient != NULL) {
            delete[] coefficient;
        }
        
        coefficient = new double[maxDegree + 1];
        for (int i = 0;i < maxDegree; ++i) {
            cout << "Enter Coeffcient " << i << ": ";
            cin >> coefficient[i];
        }
    }
    
    void polynomial::setCoEff(int power, double newCoEff){
        coefficient[power] = newCoEff;
    }
}
    polynomial polynomial::operator/(int value){
        polynomial p(maxDegree);
        for (int i =0; i < maxDegree; ++i) {
            p.setCoEff(i, getCoEff(i)/ value);
        }
        return p;
    }
    
    polynomial polynomial::operator-(){
        polynomial p(maxDegree);
        for (int i = 0 ; i < maxDegree + 1; ++i) {
            p.setCoEff(i, -1 * getCoEff(i));
            setCoEff(i, -1 * getCoEff(i));
        }
        return p;
    }
    
    polynomial polynomial::operator + (polynomial p){
        polynomial add(maxDegree > p.maxDegree ? maxDegree : p.maxDegree);
        for (int i = 0; i < maxDegree; ++i) {
            add.setCoEff(i, getCoEff(i) + p.getCoEff(i));
        }
        return add;
    }
    
    polynomial polynomial::operator - (polynomial p){
        polynomial subt(maxDegree > p.maxDegree ? maxDegree : p.maxDegree);
        for (int i = 0; i < maxDegree + 1; ++i) {
            subt.setCoEff(i, getCoEff(i) - p.getCoEff(i));
        }
        return subt;
    }
    
    ostream& operator << (ostream &out, polynomial p){
        double y;
        for (int i = p.maxDegree; i >= 0; --i) {
            y = p.getCoEff(i);
            if (y != 0) {
                if (y > 0) {
                    out << " + " << y << "*x^" << i;
                }
                else{
                    out << " " << y << "*x^" << i;
                }
            }
        }
        return out;
    }

#ifndef polynomial_h

    polynomial operator - ();

private:
    int maxDegree;
    double *coefficient;
