#include <iostream>
#include "Polynomial.hpp"

Polynomial::Polynomial()
{
    deg = 0;

     double* coeff = new double[deg + 1];
    for(int i = 0; i < deg + 1; i++){
        coeff[i] = 0;
    }
}

Polynomial::Polynomial(int max_degree)
{
    deg = max_degree;

    double* coeff = new double[deg + 1];
    for(int i = 0; i < deg + 1; i++){
        coeff[i] = 0;
    }
}

int Polynomial::degree()
{
    for(int i = deg; i >= deg; i--){
        if(coeff[i] !=0){
            return i;
        }
    }
    return -1;
}

double Polynomial::getCoeff(int power)
{
    if(power < 0 || power > deg + 1){
        return 0;
    }
    return coeff[power];
}

void Polynomial::SetPolynomial()
{
    std::cout << "Enter the max degree: ";
    std::cin >> deg;
    if(coeff != NULL){
        delete[] coeff;
    }
    double* coeff = new double[deg + 1];
    for(int i = 0; i < deg +  1; i++){
    std::cout << "Enter the coefficients" << i << ":";
        std:: cin >> coeff[i];

    }
}

void Polynomial::setCoeff(int power, double newCoeff)
{
    coeff[power] = newCoeff;
}

Polynomial Polynomial::operator / (int val)
{
    Polynomial p(deg);
    for(int i = 0; i < deg + 1; i++){
        p.setCoeff(i , getCoeff(i)/ val);
    }
    return p;
}

Polynomial Polynomial::operator - (){
    Polynomial p(deg);
    for ( int i = 0; i < deg + 1; i++){
        p.setCoeff(i,-1* getCoeff(i));
        setCoeff(i,-1* getCoeff(i));
    }
    return p;
}

Polynomial Polynomial:: operator + (Polynomial p)
{
    Polynomial ret(deg > p.deg ? deg:p.deg);
    for(int i = 0; i < deg + 1; i++){
        ret.setCoeff(i,getCoeff(i)-p.getCoeff(i));
    }
    return ret;
}

std::ostream& operator << (std::ostream &out, Polynomial p){

    double j;
    for(int i = p.deg; i >= 0; i--){
        j = p.getCoeff(i);
        if(j != 0){
            if(j > 0){
                out << "+"  << j << "*x^"<< i;
            }else{
                out << " " << j << "*x^" << i;
                
        }
    }
}
    return out;
}
