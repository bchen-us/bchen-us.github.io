#include <iostream>
#include "Polynomial.h"


//initialize polynomial to 0 by having the coefficients to 0 
Polynomial::Polynomial() {
   for(int i = 0; i < sizeof(coefficients) + 1; i++){
       coefficients[i] = 0;
   }
}

//find the highest degree
Polynomial::Polynomial(int highestDegree){
	maxDegree = highestDegree;
   	coefficients = new double[maxDegree];
   	for(int i = 0; i < maxDegree + 1; i++){
       	coefficients[i] = 0;
   }
}

//find the degree of the polynomial 
int Polynomial::degree(){
	for (int i = maxDegree; i >= maxDegree; i--) {
		if(coefficients[i] != 0) {
			return i;
    	}
    }
}

//returns the degree/power
double Polynomial::getCoeff(int power){
	if(power < 0 || power > max_degree)
       	return 0;
   	return coefficients[power];
}

//Get information from user about polynomial
void Polynomial::setPolynomial(){
	//get the maxdegree and degree of polynomial
   cout << "Enter the degree of the polynomial: ";
   cin >> maxDegree;
   
   //get coeffs from user 
   for(int i = 0; i < maxDegree + 1; ++i){
       cout << "Enter coefficient of degree " << i << ": ";
       cin >> coefficients[i];
   }
}

//set coefficient x^(power) to newCoefficient
void Polynomial::setCoefficient(int power, double newCoefficient){
   coeffs[power] = newCoefficient;
}

//overload / operator 
Polynomial Polynomial::operator / (int value){
	cout << "What scalar number would you like to divide the polynomial by?" << endl;
	cin << value;
   Polynomial q(maxDegree);
   for(int i = 0; i < maxDegree; i++){
       q.setCoeff(i, getCoeff(i) / value);
   }
   return q;
}

//overload - operator
Polynomial Polynomial::operator - (){
   Polynomial r(maxDegree);
   for(int i = 0; i < maxDegree; i++){
       r.setCoeff(i, -1 * getCoeff(i));
       setCoeff(i, -1 * getCoeff(i));
   }
   return r;
}

//overload + operator 
Polynomial Polynomial::operator + (Polynomial s) {
	//if [maxDegree > s.maxDegree] is true then return maxDegree, if not return s.maxDegree 
   Polynomial s ( (maxDegree > s.maxDegree) ? maxDegree : s.maxDegree);
   for(int i = 0; i < maxDegree; i++){
       s.setCoeff(i, getCoeff(i) + s.getCoeff(i));
   }
   return s;
}

//overload << operator
ostream& operator << (ostream &out Polynomial a) {
   double b;
   for (int i = a.maxDegree; i >= 0; i--) {
       b = a.getCoeff(i);
       if(b != 0){
           if(b > 0){
               out << " +" << b << "*x^" << i;
           }
           else{
               out << " " << b << " * x^" << i;
           }
       }
   }
   return out;
}
