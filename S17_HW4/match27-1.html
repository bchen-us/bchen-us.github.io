#include <stdio.h>

SquareMatrix::SquareMatrix(){
    initialize(10);//default is 10
}

SquareMatrix::SquareMatrix(vector<vector<double> > v2d){//initializes matrix
    //checks if matrix is a square
    
    
    bool match = true;//initializes match to true
    

    long size = v2d.size(); //initializes size to v2d.size
    
    for(int i = 0; i < size; ++i){ //for loop that runs until size of matrix
        
        if(v2d[i].size() != size){ //if size do nnot match then break
            
            match = false;
            
            break;
        }
    }
    
    if(match){//if size matchs code is executed
        
      
        
        for(int i = 0; i < v2d.size(); i++){ //for loop that runs until size of v2d.size
            
            vector<double> t;//vector t
            
            for(int j = 0; j < v2d[i].size(); j++){ //for loop that runs until size of v2d.size
                
                t.push_back(v2d[i][j]);//push back v2d[i][j] to temp
                
            }
            
            matrix.push_back(t);//push back t to matrix
        }
    }
    else{
        
        initialize(10); //initializes matrix to 10
    }
}

double SquareMatrix::getValue(int i, int j)const{//return value at position[i][j]

    
    if(i < matrix.size() && j < matrix.size() && i >= 0 && j >= 0){
        
        return matrix[i][j];
        
    }
    
    return 0;
}

void SquareMatrix::setValue(int i, int j, double value){//puts value into position [i][j]
    
    if(i < matrix.size() && j < matrix.size() && i >= 0 && j >= 0){
        
        matrix[i][j] = value; //value is put into matrix[i][j]
    }

}

void SquareMatrix::initialize(int size){  //initializes value
    

    for(int i = 0; i < size; i++){//for loop that runs until size of matrix
        
        vector<double> t; //vector t to hold value
        
        for(int j = 0; j < size; j++){//for loop that runs until size of matrix

            
            t.push_back(0.0); //default is 0.0 pushback to t
            
        }
        
        matrix.push_back(t);//push back t to matrix
    }
}


SquareMatrix SquareMatrix:: operator *(SquareMatrix &p1)const{//overloaded operator matrix 1*matrix 2

    
        
        vector<vector<double>>result; //vector result
        
    for(int i=0;i<matrix.size();i++){// for loop that runs to the size of matrix

            vector<double> d;
            
        for(int j=0;j<matrix.size();j++){// for loop that runs to the size of matrix

                d.push_back(0);//push back 0 into defult
                
            }
            result.push_back(d);//push back defult into result
 
        }
        
        
    for(int i = 0; i<matrix.size();i++){ // for loop that runs to the size of matrix

        
        for(int j=0;j<matrix.size();j++){ // for loop that runs to the size of matrix

            
                result[i][j]=0;//defult result = 0
                
            for(int k = 0;k<matrix.size();k++){// for loop that runs to the size of matrix

                
                    result[i][j]+=(matrix[i][k]*p1.matrix[k][j]);//new result = matrix 1*matix 2
                    
                }
            }
        }
        return SquareMatrix(result);
    
    return SquareMatrix();
}



SquareMatrix SquareMatrix:: operator-( SquareMatrix &p1)const  {//overloaded -
    
    
        
        vector<vector<double>>res; // vector result
        
        for(int i=0;i<matrix.size();i++){// for loop that runs to the size of matrix
            
            vector<double> d; //vector initial
            
            for(int j=0;j<matrix[i].size();j++){// for loop that runs to the size of matrix[i]

               double pushNum = (getValue(i, j) - p1.getValue(i,j)) ; //pushback vale of p1-p2
                
                d.push_back(pushNum);//pushback push num into defult
                
            }
            res.push_back(d);//push back default to result
            
        }
        
        
     
        return SquareMatrix(res);
    
    return SquareMatrix();

}




ostream& operator <<(ostream &print, SquareMatrix m){//overloaded <<
    
    
    print<<"(";
    
    for(int i = 0; i < m.matrix.size(); ++i){// for loop that runs to the size of matrix[i]

        
        for(int j = 0; j < m.matrix[i].size(); ++j){// for loop that runs to the size of matrix[i]

            
            print<< m.matrix[i][j] << " ";//prints out value at location
        }
       
        
        print<< endl;
       
    }
    
    
  print<<")"<<endl;
    
    return print;
}
