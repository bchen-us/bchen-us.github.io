LinkedList.cpp

#include<iostream>
#include<string>
#include "LinkedList.h"

using namespace std;

LinkedList::LinkedList(int max)
{
    size_linkedList = 0;
    head = NULL;
    capacity = max;
    array = new int[capacity];//array size changes with capacity=dynamic array
    size_array = 0;
}
int LinkedList::size() //gives the size of the linkedlist in an int
{
   return size_linkedList;
}

void LinkedList::push_back(int val)
{
   if(head == NULL)//if there are no values in the linkedlist
   {
       head = new Node;
       head->value = val;
       head->next = NULL;
       size_linkedList++;
       array[size_array++] = val;//changes size of array
   }
   else//if there are values in the linkedlist
   {
       struct Node *temp = head;
       struct Node *tempNode = new Node;
       tempNode->value = val;
       tempNode->next = NULL;
       while(temp->next != NULL)
       {
           temp = temp->next;
       }
       temp->next = tempNode;
       size_linkedList++;
       array[size_array++] = val;//changes size of array
   }
}

void LinkedList::pop_back()//only runs if there are values in the linkedlist
{
   if(head != NULL)
   {
       if(head->next == NULL)
           head = NULL;
       else
       {
           struct Node *temp = head;
           while(temp->next->next != NULL)
           {
               temp = temp->next;
           }
           temp->next = NULL;
       }
   }
}

ostream &operator<<( ostream &output, const LinkedList &list )//overloading the << operator
{
   struct Node *tempnode = list.head;
   int counter = 1;
   while(tempnode != NULL)
   {
       output<<counter++<<". "<<tempnode->value<<endl;
       tempnode = tempnode->next;
   }
   return output;
}

LinkedList::~LinkedList()
{
   delete head;
   delete array;
   size_array =0;
   size_linkedList = 0;
}

LinkedList.h

#ifndef LINKEDLIST_H
#define LINKEDLIST_H

using namespace std;

struct Node {
int value;
Node *next;//pointer for additional nodes
};
class LinkedList
{
   int size_linkedList; //gives the size of linkedlist
   struct Node *head;
   int *array; //makes array dynamic
   int size_array; //number of values in array
   int capacity; //the current amount of space alloted for the array
public:
   LinkedList(int);
   ~LinkedList();
   int size();
   void push_back(int);
   void pop_back();
   friend ostream &operator<<( ostream &output, const LinkedList &list );
};

#endif /* LINKEDLIST_H */
