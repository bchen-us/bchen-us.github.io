#include "LinkedList.h"
#include <iostream>
#include <string>

using namespace std;

LinkedList::LinkedList(int cap){ //constructor
	capacity = cap;
	size_linkedList = 0;
	size_array = 0;
	head = NULL;
	array = new int[capacity];
}

LinkedList::~LinkedList(){ //destructor
	//called when object ends to free up space
	Node *curr = head;
	
	while(curr != NULL){
		head = curr->next;
		delete curr;
		curr = head;
	}
}

LinkedList::LinkedList(const LinkedList& copy){ //copy constructor
	//makes a copy of an existing object
	int i = 0;
	Node* itr = NULL, *temp = NULL, *curr = NULL;

	if(copy.capacity > 0 ){
		size_array = copy.size_array;
		capacity = copy.capacity;
		array = new int[copy.capacity];
		
		while(i < copy.size_array){
			array[i] = copy.array[i];
			i++;
		}
	}
	else{
		array = NULL;
		size_array = 0;
		capacity = 0;
	}

	head = NULL;
	itr = copy.head;
	size_linkedList = copy.size_linkedList;

	while(itr != NULL){
		temp = new Node;
		temp->value = itr->value;
		temp->next = NULL;

		if(head == NULL){
			head = curr = temp;
		}
		else{
			curr->next = temp;
			curr = curr->next;
		}
		
		itr = itr->next;
	}
}

	int i = 0;
	Node* itr = NULL, *temp = NULL, *curr = NULL;

	//Delete the contents of the object's array and linked list 
	if(array != NULL)
	delete array;
	
	size_array = 0;
	capacity = 0;
    itr = head;

	while(itr != NULL){
        temp = itr;
		itr = itr->next;
		itr->next = NULL;
		delete itr;
   }
          
	size_linkedList = 0;
	itr = temp = NULL;

	if(rhs.capacity > 0 ){
		size_array = rhs.size_array;
		capacity = rhs.capacity;
		array = new int[rhs.capacity];

		while(i < rhs.size_array){
			array[i] = rhs.array[i];
			i++;
		}
	}
	else{
		array = NULL;
		size_array = 0;
		capacity = 0;
	}
  
	itr = rhs.head;
	size_linkedList = rhs.size_linkedList;

	while(itr != NULL){
		temp = new Node;
		temp->value = itr->value;
		temp->next = NULL;

		if(head == NULL){
			head = curr = temp;
		}
		else{
			curr->next = temp;
			curr = curr->next;
		}

		itr = itr->next;
	}
}


void LinkedList::pop_back(){ //removes the last element in the list
	if(head != NULL){
		if(head->next == NULL)
			head = NULL;
		else{
			struct Node *temp = head;
			
			while(temp->next->next != NULL){
				temp = temp->next;
			}
			
			temp->next = NULL;
		}	
	}

}

void LinkedList::push_back(int val){ //adds a new element at the end of the list of val
	if(head == NULL){
		head = new Node;
		head->value = val;
		head->next = NULL;
		size_linkedList++;
		array[size_array++] = val;
	}
	else{
		struct Node *temp = head;
		struct Node *tempNode = new Node;
		tempNode->value = val;
		tempNode->next = NULL;
		
		while(temp->next != NULL){
			temp = temp->next;
		}
		
		temp->next = tempNode;
		size_linkedList++;
		array[size_array++] = val;
	}
}

int LinkedList::size(){ //returns size of list
	return size_linkedList;

}

ostream &operator << (ostream &output, const LinkedList &list){ //prints out the list
	struct Node *tempnode = list.head;
	int counter = 1;
	
	while(tempnode != NULL){
		output << counter++ << "." << tempnode->value << endl;
		tempnode = tempnode->next;
	}
	
	return output;
}
