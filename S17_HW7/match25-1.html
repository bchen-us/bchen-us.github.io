#include <stdio.h>

//Destructor to deallocate memory
linkedarray::~linkedarray(){
    Node *toDel = head;
    while(toDel != NULL){
        head = toDel->next;
        delete toDel;
        toDel = head;
    }
    delete [] array;
    size_array = 0;
    size_linkedList = 0;
}

//Default constructor
linkedarray::linkedarray(){
    size_linkedList = 0;
    head = NULL;
    capacity = 100;
    try{
        array = new int[capacity];
    } catch (bad_alloc& ba){
        cerr << "bad_alloc caught: " << ba.what() << "\n";
    }
    size_array = 0;
}

//Constructor to initialize with desire capacity
linkedarray::linkedarray(int cap){
    size_linkedList = 0;
    head = NULL;
    capacity = cap;
    try{
        array = new int[capacity];
    } catch (bad_alloc& ba){
        cerr << "bad_alloc caught: " << ba.what() << "\n";
    }
    size_array = 0;
}

int linkedarray::getCap(){
    return capacity;
}

    return size_array;
}

int linkedarray::getListSize(){
    return size_linkedList;
}

}

void linkedarray::setCap(int newCap){
    capacity = newCap;
}

void linkedarray::setListSize(int newSize){
    size_linkedList = newSize;
}

// push back function to add values to Linked List and Array. If size exceeds capacity a new linked array is created and the values of original are added to it while increasing capacity.
void linkedarray::push_back(int val){

    if(head == NULL){
        try{
            head = new Node;
        }catch(bad_alloc& ba){
            cerr << "bad_alloc caught: " << ba.what() << "\n";
        }
        head->value = val;
        head->next = NULL;
        size_linkedList++;
        array[size_array] = val;
        size_array = size_array + 1;
    } else if(size_array >= capacity){

        int *temp;
        try{
            temp = new int[capacity * 2];
        } catch (bad_alloc& ba){
            cerr << "bad_alloc caught: " << ba.what() << "\n";
        }
        for(int i = 0; i < size_array; i++){
            temp[i] = array[i];
        }
        capacity = capacity * 2;
        delete [] array;
        array = temp;
        Node *tmp = head;
        Node *tmpNode;
        try{
            tmpNode = new Node;
        }catch(bad_alloc& ba){
            cerr << "bad_alloc caught: " << ba.what() << "\n";
        }
        tmpNode->value = val;
        tmpNode->next = NULL;
        while(tmp->next != NULL){
            tmp = tmp->next;
        }
        tmp->next = tmpNode;
        size_linkedList++;
        array[size_array] = val;
        size_array = size_array + 1;

    } else {
        Node *tmp = head;
        Node *tmpNode;
        try{
            tmpNode = new Node;
        }catch(bad_alloc& ba){
            cerr << "bad_alloc caught: " << ba.what() << "\n";
        }
        tmpNode->value = val;
        tmpNode->next = NULL;
        while (tmp->next != NULL) {
            tmp = tmp->next;
        }
        tmp->next = tmpNode;
        size_linkedList++;
        array[size_array] = val;
        size_array = size_array + 1;
    }
    
}

// pop back to remove elements from linked list and array. Removes the element at the end of array and linked list.
void linkedarray::pop_back(){
    if(head !=NULL){
        Node *cur = head;
        while(cur != NULL){
            cur = cur->next;
        }
        delete cur;
        size_linkedList--;
    }

    size_array--;
    int *temp;
    try{
        temp = new int[capacity];
    } catch (bad_alloc& ba){
        cerr << "bad_alloc caught: " << ba.what() << "\n";
    }
    for(int i = 0; i < size_array; i++){
        temp[i] = array[i];
    }
    delete [] array;
    array = temp;

    
}

// copy constructor for linkedarray
linkedarray::linkedarray(const linkedarray &clone1){
    Node *read = clone1.head;
    Node *tail = NULL;

    Node *toIns = NULL;
    while(read != NULL){
        try{
            toIns = new Node;
        }catch(bad_alloc& ba){
            cerr << "bad_alloc caught: " << ba.what() << "\n";
        }
        toIns->value = read->value;
        toIns->next = NULL;
        if(tail == NULL){
            head = toIns;
            tail = toIns;
        } else {
            tail->next = toIns;
            tail = toIns;
        }
        read = read->next;
    }
    capacity = clone1.capacity;
    size_array = clone1.size_array;

    size_linkedList = clone1.size_linkedList;
    try{
        array = new int[capacity];
    } catch (bad_alloc& ba){
        cerr << "bad_alloc caught: " << ba.what() << "\n";
    }
    for(int i = 0; i < clone1.size_array; i++){
        array[i] = clone1.array[i];
        }
}

//assignment operator overloaded for linked array.

linkedarray& linkedarray::operator=( const linkedarray& L1){
    if(L1.capacity == 0){
        if(array != NULL){
            delete [] array;
            array = NULL;
        }
        setArraySize(0);
    } else if(capacity == L1.capacity){
        setArraySize(L1.size_array);
        for(int i = 0; i < size_array; i++){
            array[i] = L1.array[i];
        }
    }else{
        if(array != NULL){
            delete [] array;
            array = NULL;
        }
        setCap(L1.capacity);
        setArraySize(L1.size_array);

        try{
            array = new int[capacity];
        } catch (bad_alloc& ba){
            cerr << "bad_alloc caught: " << ba.what() << "\n";
        }
        for(int i = 0; i < size_array; i++){
            array[i] = L1.array[i];
        }
    }
    
    if(L1.head == NULL){
        if(head != NULL){
            Node *toDel = head;
            while(toDel != NULL){
                head = toDel->next;
                delete toDel;
                toDel = head;
            }
        }
        head = NULL;
        setListSize(0);
    } else {
        try{
            head = new Node;
        }catch(bad_alloc& ba){
            cerr << "bad_alloc caught: " << ba.what() << "\n";
        }

        head->value = L1.head->value;
        
        Node *newPtr = head;
        for(Node *orgPtr = L1.head->next; orgPtr != NULL; orgPtr = orgPtr->next){
            try{
                newPtr->next = new Node;
            }catch(bad_alloc& ba){
                cerr << "bad_alloc caught: " << ba.what() << "\n";
            }
            newPtr = newPtr->next;
            newPtr->value = orgPtr->value;
        }
        setListSize(L1.size_linkedList);
        newPtr->next = NULL;
    }
    return *this;
}

    linkedarray::Node *tmpNode = L1.head;

    
    for(int i = 0; i < L1.size_array && i < L1.size_linkedList; i++){
        outS << "LinkedList Node: " << i <<" = " << tmpNode->value << "  Array[" << i << "]: " << L1.array[i] << "\n";
        
        tmpNode = tmpNode->next;

    }
    return outS;
}
