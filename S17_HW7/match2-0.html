#include "LinkedList.h"


LinkedList::LinkedList(int new_capacity)
{
    capacity = new_capacity;
    head = NULL;
    size_linkedList = 0;
    size_array = 0;
    array = new int[capacity];
}

void LinkedList::push_back(int val) {

    Node *a1 = NULL;
    Node *b = NULL;
    int i = 0;
    int* a = NULL;


    if (capacity == 0){

        array = new int[capacity];
        size_array = 0;
        capacity = 6;

    }

    else if (size_array == capacity) {

        a = array;
        capacity += 6;
        array = new int[capacity];

        for (i=0; i < size_array ; i++) {

            array[i] = a[i];
        }
    }

    array[size_array] = val;
    size_array++;
    a1 = head;

    if (a1 != NULL) {

        if (a1->next != NULL) {

            a1 = a1->next;
        }
    }

    b = new Node;
    b->value = val;
    b->next = NULL;

    if (a1 == NULL) {

        head = b;
    }

    else {

        a1->next = b;
    }

    size_linkedList++;
}

void LinkedList::pop_back() {

    Node *a1 = NULL;
    Node *b = NULL;

    if ((size_linkedList <= 0) && (size_array <= 0)) {

        return;
    }

    size_linkedList--;
    size_array--;

    a1 = b = head;

    if (a1->next != NULL) {

        b = a1;
        a1 = a1->next;
    }

    if (a1 == head) {

        head = NULL;
        delete a1;
    }

    else {

        b->next = NULL;
        delete a1;
    }
}

LinkedList::~LinkedList()
{
    if(array == NULL)

        delete array;

    if (head == NULL)

        delete head;

    size_array =0;
    size_linkedList = 0;
}

#ifndef CSC_HW_7_LINKEDLIST_H

#include<string>

using namespace std;

struct Node {

    int value;
    Node *next;
};

class LinkedList {

private:

    int *array; // dynamic array

    int size_array; // number of elements stored in the dynamic array

    int capacity; // the current number of integer spaces allocated to the dynamic array

    Node *head;  // point to the first node on the linked list

    int size_linkedList; // number of nodes on the linked list

public:

    LinkedList(int);

    ~LinkedList(); // destructor

    //LinkedList ( const LinkedList& clone1); // copy constructor

    //LinkedList=( const LinkedList& rhs); // overload =
