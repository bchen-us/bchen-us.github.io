#include "dynamicarray.h"

dynamicarray::dynamicarray(){
	//Array

	array = new int[3]];
	for(int i=0; i <3; i++){
		array[i]=0;
	}
	size_array=0;
	capacity = 3;
	
	//Linked List
	head-> value = 0
	size_linkedList++;
}

dynamicarray.h

#ifndef "dynamicarray.h"
#define "dynamicarray.h"



class dynamicarray{
	
	private:
		
		//DYNAMIC ARRAY
		int *array 				//dynamic array
		int size_array			//number of elements stored in the dynamic array
		int capacity			//number of int spaces allocated to dynamic array
		
		
		//LINKED LIST
		struct Node{
			int value
			Node *next
		};
		Node *head				//point to first onde on linked list
		int size_linkedList;	//number of nodes on the linked list
		
	public:
		dynamicarray();
		~dynamicarray();
		copy();
		void push(int);
		void pop()
		friend ostream& operator<<(ostream &output,const)
		
};


#endif
