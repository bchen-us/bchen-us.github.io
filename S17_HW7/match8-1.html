myarray::myarray( const myarray& clone){// copy constructor
    
    size_array = clone.size_array;// size array is cloned

        for(int i= 0; i < clone.size_array; i++){

        }
        
    } else{// else element is equal to null
        
        this->element= NULL;// else element is equal to null
        
    }
    size_linkedList=clone.size_linkedList;// linked list is cloned
    
    if(clone.head==NULL){// check if head is null
        head=NULL;// head is set to null
    } else{//else code is ran
        head = new Node;// head is set to new node
        head->value= clone.head->value;// head value is set to clone head value
        Node *New = new Node;// new is set to node
        New = head;// new is set to head

}

myarray myarray::operator=(const myarray& rhs){//overloaded =
    
    if(rhs.capacity ==0){// if statement to delet element

            delete[] element;// delete element
            
            element=NULL;// element is set to null
            
        }
        
        size_array=0;// size array is set to 0
        
    }
    else if (capacity==rhs.capacity){// check if capacity is equal to rhs. capacity
        size_array= rhs.size_array;// sizee is to rhs. size
        for(int i=0; i<size_array; i++){// for loop that runs through size array
             element[i]=rhs.element[i];// element is then set
        }
        
    }
    else{
        if(element!=NULL) {// if element is null
            delete[] element;// element is deleted
            element = NULL;// element is set to null
        }
        
        capacity=rhs.capacity;// capacity is clones
        
        size_array=rhs.size_array;// size array is cloned
        
        element= new int [capacity];
        for(int i = 0; i<size_array; i++){// for loop that runs through size array
            
            
            element[i]=rhs.element[i];// element is copied
        }
        
    }
   
    
    if(rhs.head==NULL){// check if head is = to null
        
        if(head != NULL){// if head is not equal to null continue
            
            Node *index = head;// idex is set to head
            
            while(head!=NULL){// while head is not null
                
                head = index->next;// head is set to index
                
                delete index;// index is deleted
                
                index = head;//index is set to head
            }
        }
        head=NULL;// head is set to null
        size_linkedList=0;// size is set to 0
    }else{
        head = new Node;// head is set to node
        head->value= rhs.head->value;
        Node *New = new Node;// new is set to node
        New = head;// new is set to head
        for(Node *i = rhs.head->next; i !=NULL; i=i->next){// for loop
            New->next= new Node;// new nex is set ot node
            New = New-> next;// new is set to new next
            New->value= i->value;
            
        }
        New->next=NULL;
        
    }
    return *this;// this is returned
}

void myarray::push_back(int k){// push back function
    

    if(size_array>=capacity){// if statement to ceck if size array is = to capacity
        
        
        capacity=capacity+1; // capacity is raised by one
        
        int *temp= new int[capacity];// temp is se to in capacity
        
        for(int i=0; i <size_array;i++){//for loop that runs through size array
            
            temp[i]=element[i]; // temp is set to element
        }
        delete [] element;// element is deleted
        
        size_array++;// move one to next value
        
        element=temp;// element is set to temp
        
        int i= size_array-1; // i is set to 1 less then size array
        
        element[i]=k;// element is pushed back
        
    }else{
        size_array++;// size array is increaded by one
        int i = size_array-1;// i is size array minus one
        element[i]=k;// element is pushed back
    }
    if(head==NULL){// check if head is null
        
        head = new Node;// head is set to node
        
        head->value=k;// head value is set to k
        
        head->next=NULL;// head next is set to null
        
        size_linkedList++;// size linked list is increased by one
        

    }else{
        
        Node *dNode= new Node;// dnode is set to new node
        
        dNode->value=k;//dnode valuse is set to k
        
        dNode->next=NULL;// dnode next is set to null
        
        Node *tep=new Node;// tep is set to node
        
        tep=head;// tep is set to head
        
        while(tep->next != NULL){// hwat tep is not null code is ran
            
            tep= tep->next;
        }
        tep->next=dNode;
        size_linkedList++;// size linked list is increaded by one

    }
}

void myarray:: pop_back(){// pop back function
    
    Node *p =new Node;// p ist set to node
    
    if(head != NULL){// check is head is not set to null
        
        if(head->next == NULL){
            
            p = head;//p is set to head
            
            head = head->next;// head is set to head next
            
        }else{
        
        Node *tmp= new Node; //temp is set to node
        
        tmp= head;// tmp is set to head
        
        for(int i=1;i<size_linkedList-1;i++){// for loop that runs through the size of linked list
            
            tmp=tmp->next;// temp is set to tmp next
            
        }
            
        p=tmp->next;
        
        tmp->next= p->next;
        
    }
    p->next=NULL;
    
    delete p;// p is deleted
    
    p=NULL;// p is set to null
    
    int *array =new int[capacity-1];// array is set to capacity - 1
    
    for(int i=0;i<capacity-1;i++){// for loop that runs though capacity-1
        
        array[i]=element[i];// array i is set to element i
        
    }
    delete [] element;// element is deleted
    
    element=array;// element is equal arrray
    
    capacity--;// capacity is decreased by 1
    
    size_array--;// size array is decreaded by one
    
    size_linkedList--;// size linked list is decreased by one
    }
}

#ifndef array_h
#define array_h
#include <iostream>

using namespace std;


class myarray{
    
private:
    
    int *element; // dynamic array
    
    int size_array;// # of elements stored in dynamic array
    
    int capacity;// intger spaces allocated to dynamic array
    
    struct Node{
        
        int value;
        
        Node *next;
        
    };
    
    Node *head;
    
    int size_linkedList;//link list

public:
    
    ~myarray(); //destructor
    
    myarray(const myarray& clone);//copy constructor
    
    myarray operator=(const myarray& rhs);// overloaded
    
    myarray();
    
    
    void push_back(int k); //push back
    
    void pop_back();// pop back
    
    
    friend std::ostream& operator<<(std::ostream& out,const myarray p);// overloaded put operator
  
    


};
