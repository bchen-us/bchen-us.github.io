using namespace std;


ADTSinglyLinkedList::ADTSinglyLinkedList()
{
    cap = SIZE; //the  number of int spaces allocated for the dynamic array
    array=new int[cap]; //dynamic array
    sizeArray = 0; //number of elements stored in the dynamic array

    head = NULL; //point to the first node on the linked list
    sizeLinkedList = 0; //number of nodes on the linked list
}

ADTSinglyLinkedList::~ADTSinglyLinkedList()
{
    delete (array);
    delete (head);
}

void ADTSinglyLinkedList::push_back(int val )
{
    if(sizeArray == cap)
    {
        //increase array capacity by GrowBy
        int *temp = new int[cap + SIZE];
        for(int i = 0; i < sizeArray; i++)
        {
            temp[i] = array[i];//copy items to new array
        }
        array = new int[cap + SIZE];//assign new array to old array
        for(int i = 0; i < sizeArray; i++)
        {
            array[i] = temp[i];//copy items from new array
        }
        cap += SIZE;
        delete (temp);//delete new array
    }
    array[sizeArray++] = val;//insert value in array

    //insert value in linked list
    if(head == NULL)
    {
        head=new Node();
        head-> value = val;
        head-> next = NULL;
    }
    else
    {
        Node *temp = head;
        while(temp -> next != NULL)//FINDING LAST NODE YO
            temp = temp -> next;
        temp-> next = new Node();//insert into the last
        temp-> next -> value=val;
        temp-> next -> next = NULL;
    }
    sizeLinkedList++;

}

void ADTSinglyLinkedList::pop_back()
{
    Node* temp = NULL, *prev = NULL;
    if((sizeArray <= 0) && (sizeLinkedList <= 0))
    {
        return;
    }

    /*I will not deallocate value in the array, rather i will decrement the count*/
    sizeArray--;

    sizeLinkedList--;
    temp = prev = head;
    while(temp -> next != NULL)
    {

        prev = temp;
        temp = temp->next;
    }

    if(temp == head)
    {
        head = NULL;
        delete temp;
    }
    else
    {
        prev -> next = NULL;
        delete temp;
    }

}

ostream& operator<<(ostream& out, const ADTSinglyLinkedList &obj)
{
    out <<"Array: ";
    for(int i=0; i<obj.sizeArray; i++)
        out<<obj.array[i]<<" "; //print array
    out<< endl << "Linked list: ";
    Node *temp = obj.head;
    for(int i=0; i<obj.sizeLinkedList; i++) //FIND LAST NODE
    {
        out << temp -> value <<" "; //print linked list
        temp = temp -> next; //get next pointer
    }
    out << endl;
    return out; //return ostream
}

#define SIZE 100


struct Node {

};
class ADTSinglyLinkedList
{
   private:
       int *array; //dynamic array
