#include <iostream>
#include "list.h"
#include <stdlib.h>

using namespace std;

llist::llist(int capacity)
{
	sizellist = 0;
	hd = NULL;
	cap = capacity;
	arr = new int[cap];
	sizeA = 0;

}

void llist::push_back(int value)
{
	if(hd == NULL)
	{
		hd = new Node;
		hd->val = value;
		hd->nxt = NULL;
		sizellist++;
		arr[sizeA++] = val;
	}
	else
	{
		struct Node *tmp = hd;
		struct Node *tmpNode = new Node;
		tmpNode->val = value;
		tmpNode->nxt = NULL;
		while(tmp->nxt != NULL)
		{
			tmp = tmp->nxt;
		}
		tmp->nxt = tmpNode;
		sizellist++;
		arr[sizeA++] = value;
	}
}

void llist::pop_back()
{
	if(hd != NULL)
	{
		if(hd->nxt == NULL)
			hd = NULL;
		else
		{
			struct Node *tmp = hd;
			while(tmp->nxt->nxt != NULL)
			{
				tmp = tmp->nxt;
			}
			tmp->nxt = NULL;
		}
	}
}

int llist::size()
{
	return sizellist;
}


ostream &operator<<( ostream &output, const llist &list )
{
	struct Node *tmpnode = list.hd;
	int count = 1;
	while(tmpnode != NULL)
	{
		output<<count++<<". "<<tmpnode->val<<endl;
		tmpnode = tmpnode->nxt;
	}
	return output;
}

llist::~llist()
{
	delete hd;
	delete arr;
	sizeA =0;
	sizellist = 0;
}

#ifndef list_h
#define list_h
#include<iostream>

using namespace std;


struct Node {
	int val;
	Node *nxt;
};

class llist
{
	int sizellist;
	struct Node *hd;
		int *arr;
		int sizeA;
		int cap;
	public:
	llist(int);
	~llist();
	
	void push_back(int);
	
	void pop_back();
	
	friend ostream &operator<<( ostream &output, const llist &list );
	
	int size();
};
