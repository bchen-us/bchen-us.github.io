#include <iostream>
#include "LinkedList.h"

using namespace std;

LinkedList::LinkedList(){
	size_linkedList=0;
	head = NULL;
	array=NULL;
	size_array=0;

	capacity=0;
};

LinkedList::~LinkedList(){
	if(array != NULL){
		delete[] array;
	}
	size_array=0;
	
	Node *del;
	while(del != NULL){
		head = del->next;
		delete del;
		del=head;
	}
	size_linkedList=0;
}

LinkedList::LinkedList(const LinkedList& lhs){
	if(lhs.array != NULL){
		array = new int[lhs.size_array];
		size_array = lhs.size_array;
		for(int i =0;i<lhs.size_array; i++){
			array[i]=lhs.array[i];
		}
	}
	
	Node* read = lhs.head,*tail=NULL, *ins=NULL;
	while(read != NULL){
		ins=new Node;

		ins->value= read->value;
		ins->next=NULL;
		if(tail==NULL){
			head=tail;
			tail=ins;
		}else{
			tail->next=ins;
			tail=ins;
		}
		read=read->next;
	}
	size_linkedList=lhs.size_linkedList;

}

LinkedList LinkedList::operator=(const LinkedList& lhs){
	if (lhs.capacity == 0){
        if(array != NULL) {
        	delete[] array;
       }

       size_array = 0;
    } else if (capacity == lhs.capacity) {
    	size_array = lhs.size_array;
    	for(int i=0; i < size_array; i++){
    		array[i]=lhs.array[i];
		}
    } else {
        if(array != NULL) {
        	delete[] array;
        }
        capacity = lhs.capacity;
        size_array = lhs.size_array;
        
        array = new int[capacity];
        for(int i=0; i < size_array; i++){
    		array[i]=lhs.array[i];
		}
    }
    
	Node *temp;
	if(lhs.head == NULL){
		if(head != NULL) {
           	while(head != NULL){
				head->next = temp->next;
	    		temp->next = NULL;
    			delete temp;
    			temp = head->next;
			}
        }
        head = NULL;
        size_linkedList = 0;
    } else{
    	head = new Node;
    	head = lhs.head;
    	for(int i =0; i<size_linkedList;i++){
    		head->value=lhs.head->value;
    		head->next = lhs.head->next;
		}
	}
}

void LinkedList::push_back(int val){
	int *temp = new int[capacity+1];
	for(int i= 0; i<capacity;i++){
		temp[i] = array[i];
	}
	capacity++;
	size_array++;
	delete[] array;
	array=temp;
	array[capacity-1]=val;
	
	
	Node *curr= new Node;
	curr->value=val;
	curr->next=head;
	head=curr;

	size_linkedList++;
}

void LinkedList::pop_back(){
	int *temp = new int[size_array-1];
	
	for(int i =0; i<size_array-1; i++){
		temp[i]=array[i];
	}
	capacity--;
	size_array--;
	delete[] array;
	array=temp;
	
	if(head != NULL){
		head = head->next;

		size_linkedList--;
	}
	
}

ostream& operator<<(ostream& out, const LinkedList& rhs){
	for(int i=0; i<rhs.size_array;i++){
		out << rhs.array[i] <<", ";
	}
	out<<endl;
	
	LinkedList::Node *temp =rhs.head;
	for(int i = 0; i<rhs.size_linkedList;i++){
		
		out << temp->value<<", ";
		temp= temp->next;
	}
	out <<endl;
	
	return out;
}

#ifndef LinkedList_h
#define LinkedList_h

using namespace std;


class LinkedList{
	private:
		int* array; //dynamic array
		int size_array; //nummber of elements stored in the dynamic array
		int capacity; //current number of integer spaces allocated in the dynamic array
		struct Node{
			int value;
			Node *next;
		};	
		Node *head; //point to the first Node on the linked list
		int size_linkedList; //number of nodes on the linked list
		
	public:
		LinkedList();
		~LinkedList();
		LinkedList(const LinkedList&);
		LinkedList operator=(const LinkedList&);
		
		void push_back(int val);
		void pop_back();
		friend ostream& operator<<(ostream&, const LinkedList&);
};
#endif
