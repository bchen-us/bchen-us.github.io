#include<iostream>
using namespace std;
#define SIZE 100
struct Node {
   int value;
   Node *next;
};
class ADTlist
{
private:
	int *array; // a dynamic array
	int size_array; // the number of elements stored in the dynamic array
	int capacity; // the current number of integer spaces allocated to the dynamic array

	Node *head; // points to the first node on the linked list
	int size_linkedList; // the number of nodes on the linked list
public:
	ADTlist();
	~ADTlist();
	void push_back(int val);
	void pop_back();
	friend std::ostream& operator<<(std::ostream& out, const ADTlist &obj);
};

#include "ADTheader.h"
using namespace std;

ADTlist::ADTlist() {
    capacity = SIZE; //the current number of integer spaces allocated to the dynamic array
    array = new int[capacity]; //dynamic array
    size_array = 0; //number of elements stored in the dynamic array  

    head = NULL; //point to the first node on the linked list
    size_linkedList = 0; //number of nodes on the linked list
}

ADTlist::~ADTlist() {
    delete(array);
    delete(head);
}

void ADTlist::push_back(int val) {
    if (size_array == capacity) {
        //increase array capacity by GrowBy
        int *temp = new int[capacity + SIZE];
        for (int i = 0; i < size_array; i++) {
            temp[i] = array[i]; //copy items to new array
        }
        array = new int[capacity + SIZE]; //assign new array to old array
        for (int i = 0; i < size_array; i++) {
            array[i] = temp[i]; //copy items from new array
        }
        capacity += SIZE;
        delete(temp); //delete new array
    }
    array[size_array++] = val; //insert value in array

    //insert value in linked list
    if (head == NULL) {
        head = new Node();
        head->value = val;
        head->next = NULL;
    } else {
        Node *temp = head;
        while (temp->next != NULL)//FIND LAST NODE
            temp = temp->next;
        temp->next = new Node(); //insert at last
        temp->next->value = val;
        temp->next->next = NULL;
    }
    size_linkedList++;

}

void ADTlist::pop_back() {
    size_array--;
    Node *temp = head;
    for (int i = 0; i < size_linkedList - 1; i++)//FIND LAST NODE
        temp = temp->next;
    delete(temp->next);
    temp->next = NULL; //delete node
    size_linkedList--;
}

ostream& operator<<(ostream& out, const ADTlist &obj) {
    out << "Array: ";
    for (int i = 0; i < obj.size_array; i++)
        out << obj.array[i] << " "; //print array
    out << endl << "Linked list: ";
	Node *temp = obj.head;
    for (int i = 0; i < obj.size_linkedList; i++)//FIND LAST NODE
    {
        out << temp->value << " "; //print linked list
        temp = temp->next; //get next pointer
    }
    out << endl;
    return out; //return ostream

}
