#include "LinkedList.h"

//default constructor
LinkedList::LinkedList():size_array(0),head(NULL)
{

}

//destructor
LinkedList::~LinkedList()
{

    Node *toDel = head;
    while(toDel!=NULL)
    {
        head = toDel->next;
        delete toDel;
        toDel = head;
    }
}

//copy constructor
LinkedList::LinkedList(const LinkedList& rhs)
{
    *this=rhs;
}

//overloaded assignment operator
LinkedList LinkedList::operator=(const LinkedList& rhs)
{
    if(rhs.head==NULL)
    {
        head=NULL;
        size_array=0;
    }
    else
    {
        Node *curr = rhs.head;
        Node *rhsCurr = rhs.head;
        Node *rhsHead = rhs.head;
        while (rhs.head!=NULL)
        {
            head->value=rhsHead->value;
            head=curr->next;
            rhsHead=rhsCurr->next;
            curr=head;

            rhsCurr=rhsHead;
        }
    }
    return *this;
}

void LinkedList::pushback(int val)
{
    if(head==NULL)
    {
        Node *newNode = new Node;
        newNode->value=val;
        newNode->next=NULL;
        head=newNode;
        size_linkedList++;
    }
    else
    {
        Node *newNode = new Node;
        newNode->value=val;
        newNode->next=NULL;
        Node *curr=head;
        Node *head2=head;
        while(curr->next!=NULL)
        {
            curr=head2->next;
            head2=curr;
        }
        curr->next=newNode;
        size_linkedList++;
    }
}

void LinkedList::popback()
{
    if(head==NULL)
        cout<<"List is empty";
    else if(head->next==NULL)
        delete head;
    else
    {
        Node *curr=head;
        Node *prev=head;

        while(curr->next!=NULL)
        {
            prev=curr;
            curr=curr->next;
        }
        delete curr;
        prev->next=NULL;
        size_linkedList--;
    }
}
ostream &operator<<(ostream &output, const LinkedList &arr)
{
    LinkedList::Node* LList=arr.head;
    while(LList!=NULL)
    {
        output<<LList->value<<" ";
        LList=LList->next;
    }
    return output;
}
