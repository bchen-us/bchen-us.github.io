//to identify the most frequent letter
char getMostFreqLetter(string str)
{
    int max=0;
    int count=0;
    char maxFreCharcter = ' ';
    
    // Convert all letters into lowercase
    for (int i=0; str[i]; i++){
        str[i] = tolower(str[i]);
    }
    
    // letter between 'a' to 'z'
    for(char letter='a'; letter<='z'; letter++)
    {
        count = 0;
        for(int i=0; i<str.length();i++)
        {
            if(str[i] == letter)
                count++;
        }
        
        if(count>max)
        {
            max=count;
            maxFreCharcter=letter;
        }
    }
    
    return maxFreCharcter;
}

//To identify and print all the tokens contained in s on the standard output
void StrToTokens(string str, char ch)
{
    vector<string> res;
    
    str += ch;
    // Start point
    int startPoint = 0;
    // End point of delimiter
    int lastPoint = str.find(ch);
    
    while (lastPoint < str.size())
    {
        if (startPoint != lastPoint)
            res.push_back(str.substr(startPoint, lastPoint - startPoint));
        // Update start point
        startPoint = lastPoint + 1;
        // Move to next
        lastPoint = str.find(ch, startPoint);
    }
    
    // Output the result
    for(int i=0; i<res.size(); i++)
        cout<< res[i] << endl;
}
