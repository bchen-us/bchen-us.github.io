#include<iostream>
#include<string>
#include<vector>
#include <sstream>
using namespace std;
struct TokenFreq
{
    string token;
    int freq;
};
void selectionSort_v1(vector<TokenFreq> & tokFreqVector)
{
    TokenFreq temp;
    int min;
    
    for(int i = 0; i < tokFreqVector.size()-2; i++){
        min = i;
        for (int j = i+1; j <tokFreqVector.size()-1; j++){
            if (tokFreqVector[j].freq < tokFreqVector[min].freq)
                min = j;
        }
        if(min != i)
        {
            temp = tokFreqVector[i];
            tokFreqVector[i] = tokFreqVector[min];
            tokFreqVector[min] = temp;
        }
    }
    cout<<"-------Selection Sort-------"<<endl;
    for(int i = 0; i<tokFreqVector.size()-1; i++)
    {
        cout<<tokFreqVector[i].token<<": "<<tokFreqVector[i].freq<<endl;
    }
}

vector<TokenFreq> getTokenFreq(string & s)
{
    int ind = 0;
    vector<TokenFreq> tokFreq;
    string term = "";
    tokFreq.push_back(TokenFreq());
    int cnt = 0;
    int marker = 0;
    for(int k = 0; k < s.length(); k++)
    {
        if(s[k] ==' ' || s[k] ==',' || s[k] =='.' ||s[k] ==';' || s[k] ==':')
        {
            marker = 0;
            for(int i = 0; i < tokFreq.size(); i++)
            {
                if(tokFreq[i].token == term)
                {
                    
                    tokFreq[i].freq++;
                    marker = 1;
                    term = "";
                }
            }
            if(marker == 0)
            {
                cnt++;
                tokFreq[ind].token = term;
                tokFreq[ind].freq = 1;
                ind++;
                tokFreq.push_back(TokenFreq());
                term = "";
            }
        }
        else
        {
            term=term+s[k];
        }
    }
    marker = 0;
    for(int i = 0; i<tokFreq.size(); i++)
    {
        if(tokFreq[i].token == term)
        {
            tokFreq[i].freq++;
            marker = 1;
            term = "";
        }
    }
    if(marker == 0)
    {
        cout<<term;
        tokFreq[ind].token =term;
        tokFreq[ind].freq = 1;
        ind++;
    }
    cout<<"-------Token frequency-------"<<endl;
    
    for(int i = 0; i<tokFreq.size()-1; i++)
    {
        cout<<tokFreq[i].token<<", "<<tokFreq[i].freq<<endl;
    }
    return tokFreq;
}
