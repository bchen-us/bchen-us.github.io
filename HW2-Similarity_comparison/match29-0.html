#include <iostream>
#include <string>
#include <string.h>
#include <vector>
using namespace std;

struct TokenFreq {
	string token;
	int freq;

};

vector<TokenFreq> getTokenFreq(string& s) {
	vector <TokenFreq> structVector; //This is the vector that I will be returning. It stores Struct objects in it
	TokenFreq structObject= {"",0}; //This is used to initialize structObjects that I will be pushing into the vector
	int structVectorSize=0; //This is the size of the vector, it will increase each time I push something into it
	bool uniqueToken=true; //This is the boolean reads true if the token is not currently in the in the vector
	//this outer loop will loop through the entire string
	for(int a=0; a<s.length(); a++) {
		//when the program finds a space in the string it will see if the current token matches any tokens already in the structVector
		if(s[a]==' ') {
			uniqueToken=true;
			for(int b= 0; b<structVectorSize; b++) {
				//if the program finds a repeated token it only increases the frequency at that specific element in the structVector
				//it then sets the token to "" so that it can write a new token into it
				if(structObject.token==structVector[b].token) {
					structVector[b].freq++;
					uniqueToken=false;
					structObject.token="";
				}//end if
			}//end for b
			//if it does find a token that is not in the vector, it will make its frequency 1 and push the entire structObject into the vector
			//the token is then reset to "" and the size of the Vector increases
			if(uniqueToken==true) {
				structObject.freq=1;
				structVector.push_back(structObject);
				structObject.token="";
				structVectorSize++;
			}//end if

		}//end if s[a]==' '
		else {
			//This is pushing each character in the string into a StructObject.token
			structObject.token.push_back(s[a]);
		}
	}//end for a
	//since there is no space after the last word in the string, this finds the last word and does the same process as above
	uniqueToken=true;
	for(int b=0; b<structVectorSize; b++) {
		if(structObject.token==structVector[b].token) {
			structVector[b].freq++;
			uniqueToken=false;
			structObject.token="";
		}//end if
	}//end for b
	if(uniqueToken==true) {
		structObject.freq=1;
		structVector.push_back(structObject);
		structObject.token="";
		structVectorSize++;
	}//end if
	cout<<endl;
	cout<< "The contents of struct vector are: "<< endl;
	for(int x=0; x<structVector.size(); x++) {
		cout<< "{" << structVector[x].token << " " << structVector[x].freq << "}," << " ";
	}
	cout<<endl;

	return structVector;
}

void selectionSort_v1(vector<TokenFreq> & tokFreqVector) {
	//The essential idea behind this is that when we pass in a vector<TokenFreq>
	//this will just take the first element in tokFreqVector, compare it to the one next to it, and if it is smaller it swaps them
	for(int x=0; x<tokFreqVector.size(); x++) {
		for(int y=0; y<tokFreqVector.size(); y++) {
			if(tokFreqVector[x].freq>=1) {
				if(tokFreqVector[x].freq<tokFreqVector[y].freq) {
					swap(tokFreqVector[y].token, tokFreqVector[x].token);
					swap(tokFreqVector[y].freq, tokFreqVector[x].freq);
				}//end if
			}//end if
		}//end for y
	}//end for x
	cout<<endl;
	cout<< "The sorted tokFreqVector is: " << endl;
	for(int a=0; a<tokFreqVector.size(); a++) {
		cout << tokFreqVector[a].token << " " << tokFreqVector[a].freq <<endl;
	}

}//end void selectionSort_v1

vector<TokenFreq>& selectionSort_v2(vector<TokenFreq> tokFreqVector ) {
	for(int x=0; x<tokFreqVector.size(); x++) {
		for(int y=0; y<tokFreqVector.size(); y++) {
			if(tokFreqVector[x].freq>=1) {
				if(tokFreqVector[x].freq<tokFreqVector[y].freq) {
					swap(tokFreqVector[y].token, tokFreqVector[x].token);
					swap(tokFreqVector[y].freq, tokFreqVector[x].freq);
				}//end if
			}//end if
		}//end for y
	}//end for x
	cout<<endl;
	cout<< "The sorted tokFreqVector_V2 is: " << endl;
	for(int a=0; a<tokFreqVector.size(); a++) {
		cout << tokFreqVector[a].token << " " << tokFreqVector[a].freq << endl;
	}
	return tokFreqVector;
}//end vector<TokenFreq>& selectionSort_v2
