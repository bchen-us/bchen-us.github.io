vector <tokenfreq> gettokenfreq ( string & s)
{
    tokenfreq token;
    vector <tokenfreq> tokenvec;
    string temp[100];                     // temp strings to store tokens
    int k = 0;                           //counter for temp strings
    for (int i = 0 ; i <s.size() ; i ++)
    {
        if(s[i]!=' ')                    // go through string by each char
        {
            temp[k] += s[i];             // no space  store the char in first temp string
        }
        else
        {
            k++;                         // reach space  switch to the next temp string
        }
    }

    int freq[100]; for (int q = 0 ; q < 100; q ++) { freq[q] = 0 ;} // frequency of tokens
    for ( int s = 0 ; s < k+1; s ++)
    {
        for (int l = 0 ; l < k+1; l ++)
        {
            if ( temp[s] == temp[l])      // 2 tokenss match
            {
                freq[s] ++;               //its frequency ++
            }
        }
    
    }
    
    
    for ( int j = 0 ; j < k+1; j ++)
    {
        token.token = temp[j];          //set token and freq
        token.freq = freq[j];
        tokenvec.push_back(token);         //then add it to the vector
    }
    
    return tokenvec;
}

// selection sort  copied from the website
void selectionSort_v1( vector<tokenfreq> & tokFreqVector )
{
    int i,j;
    
    for (j = 0; j < tokFreqVector.size()-1; j++) {
        int iMin = j;
        for ( i = j+1; i < tokFreqVector.size(); i++) {
            if (tokFreqVector[i].freq < tokFreqVector[iMin].freq) {
                iMin = i;
            }
        }
        
        if(iMin != j) {
            swap(tokFreqVector[j], tokFreqVector[iMin]);
        }
    }
}

// selection sort  copied from the website
vector<tokenfreq>& selectionSort_v2(  vector<tokenfreq>  tokFreqVector )
{
    int i,j;
    for (j = 0; j < tokFreqVector.size()-1; j++) {
        int iMin = j;
        for ( i = j+1; i < tokFreqVector.size(); i++) {
            if (tokFreqVector[i].freq < tokFreqVector[iMin].freq) {
                iMin = i;
            }
        }
        
        if(iMin != j) {
            swap(tokFreqVector[j], tokFreqVector[iMin]);
        }
    }
    
    return tokFreqVector;
}
