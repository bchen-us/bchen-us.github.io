struct TokenFreq
{
	// to store specific token and frequency
	string token;
	int freq;
};
//vector of TokenFreq objects by refrence and applies the selection
//sort to sort the vector in increasing order of token frequences

void selectionSort_v1(vector<TokenFreq> & tokFreqVector)
{
	TokenFreq tmp;
	int min;

	for (int i = 0; i < tokFreqVector.size() - 2; i++) {
		min = i;
		for (int j = i + 1; j <tokFreqVector.size() - 1; j++) {
			if (tokFreqVector[j].freq < tokFreqVector[min].freq)
				min = j;
		}
		if (min != i)
		{
			tmp = tokFreqVector[i];
			tokFreqVector[i] = tokFreqVector[min];
			tokFreqVector[min] = tmp;
		}
	}
	cout << "Tokens and its Frequency in Increasing Order" ".\n";
	cout << "......................" ".\n" << endl;
	for (int i = 0; i<tokFreqVector.size() - 1; i++)
	{

		cout << tokFreqVector[i].token << ": " << tokFreqVector[i].freq << endl;
	}
}
