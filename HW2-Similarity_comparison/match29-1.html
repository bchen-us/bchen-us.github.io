#include <iostream>

#include <string.h>
#include <vector>
using namespace std;


/*
this is the struct used to store the token and the frequency of said token
*/

struct TokenFreq
{
	string token;
	int freq;
};
/*
this is the function used to take a string and output a vector of scructs
*/
vector<TokenFreq> getTokenFreq(string& s)
{
	vector<TokenFreq> tokenVector; //initializes the vector of struts
	TokenFreq tokenFreq = {"",0}; // initializes the data members of the vector
	int sizeVector = 0; //checks how many unique entries are in the vector, therefore can know how many times to check for duplicates
	bool notFound = true; //checks if a duplicate is found
	
	for(int spotInString = 0; spotInString<s.length();spotInString++)  //iniital for loop going through each character in the string
	{
		if(s[spotInString]== ' ') //if a space is found, detects a new word
		{
			notFound = true;
			for(int tokenFreqScanner = 0; tokenFreqScanner < sizeVector;tokenFreqScanner++) //scans the vector to find out if there are duplicates
			{
				if(tokenFreq.token==tokenVector[tokenFreqScanner].token) 
				{
					tokenVector[tokenFreqScanner].freq++;
					notFound = false;
					tokenFreq.token = "";
				}
					
			}
			if(notFound == true) //creates new entry if no duplicate was found
			{
					tokenFreq.freq=1;			
					tokenVector.push_back(tokenFreq);
					tokenFreq.token = "";
					sizeVector++;	//incresases how many unique entries in the vector			
			}
				
		}
		else
		{
		tokenFreq.token.push_back(s[spotInString]); //adds character by character until a space is found
		}
	}
	//at the end of the string, will run the loop for the vector for the last word.
	notFound = true;
			for(int tokenFreqScanner = 0; tokenFreqScanner < sizeVector;tokenFreqScanner++) //scans the vector to find out if there are duplicates
			{
				if(tokenFreq.token==tokenVector[tokenFreqScanner].token) 
				{
					tokenVector[tokenFreqScanner].freq++;
					notFound = false;
					tokenFreq.token = "";
				}
					
			}
			if(notFound == true) //creates new entry if no duplicate was found
			{
					tokenFreq.freq=1;			
					tokenVector.push_back(tokenFreq);
					tokenFreq.token = "";
					sizeVector++;				
			}
	

	return tokenVector; // returns the vector of struts
}
/* This will pass the vector as an argument, and sort that array */
void selectionSort_v1(vector<TokenFreq> & tokFreqVector)
{
	TokenFreq temp;
	for(int i = 0;i<tokFreqVector.size()-1;i++)
	{
		for(int j = i+1; j < tokFreqVector.size();j++)
		{
			if(tokFreqVector[i].freq > tokFreqVector[j].freq)
			{
				temp=tokFreqVector[i];
				tokFreqVector[i]=tokFreqVector[j];
				tokFreqVector[j]=temp;
			}
			
		}
	}
		
}
/* This will sort the array and return a new array */
vector<TokenFreq> selectionSort_v2(vector<TokenFreq> tokFreqVector)
{
	vector<TokenFreq> newVector;
	TokenFreq temp;
	for(int copy = 0; copy<tokFreqVector.size();copy++) //makes a copy of the array
		newVector.push_back(tokFreqVector[copy]);

	for(int i = 0;i<newVector.size()-1;i++)
	{
		for(int j = i+1; j < newVector.size();j++)
		{
			if(newVector[i].freq > newVector[j].freq)
			{
				temp=newVector[i];
				newVector[i]=newVector[j];
				newVector[j]=temp;
			}
			
		}
	}
	return newVector;
	
	
}
