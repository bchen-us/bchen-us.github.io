#include<iostream>
#include<vector>
using namespace std;

struct TokenFreq
{
	string token; 
	int freq;
};



 vector<TokenFreq> getTokenFreq(string& s)
 {
	int index =0;
  	vector<TokenFreq> r;
	r.push_back(TokenFreq());

   	string words="";
   	int f=0;
   	int flag=0;
   
   for(int k=0;k<s.length();k++)
   {
       if(s[k]==' ' || s[k]==',' || s[k]=='.' ||s[k]==';' || s[k]==':' || s[k]=='?' || s[k] =='!')
       {
           flag=0;
           for(int i = 0; i<r.size(); i++)
           {
               if(r[i].token==words)
               {
                	r[i].freq++;
                   flag=1;
                   words="";
               }
            }
           if(flag==0)
            {
                f++;
                r[index].token=words;
                r[index].freq=1;
                index++;
                r.push_back(TokenFreq());     
                words="";
            }
       }
       else
       {
           words=words+s[k];
       }
   }
  
   flag=0;
   for(int i = 0; i<r.size(); i++)
    {
        if(r[i].token==words)
        {
            r[i].freq++;
            flag=1;
            words="";
        }   
    }
   	if(flag==0)
	{
    	cout<<words;
   		r[index].token=words;
   		r[index].freq=1;
   		index++;
   }
   string t = "\n-------------------------------------------------------------------------------\n";
   string bSort = "Before Sort (Format {Vectors : Frequency},) ";
   cout << t  <<  bSort << t << endl << "{" ;

   for(int i = 0; i<r.size()-1; i++)
   {    	
       cout<<r[i].token<<": "<<r[i].freq<<", ";
   }
  	cout <<"}" << endl;
  	
   	return r;
}



vector<TokenFreq>& selectionSort_v1( vector<TokenFreq> tokFreqVector )
{
	TokenFreq temp;
    int min;
    string sorted = "Sorted Array (Format {Vectors : Frequency},)";
    string t = "\n-------------------------------------------------------------------------------\n";
    


	for( int i = 0; i < tokFreqVector.size()-2; i++){
		min = i;
		for (int j = i+1; j <tokFreqVector.size()-1; j++){
			if (tokFreqVector[j].freq < tokFreqVector[min].freq)
				min = j;
		}
    	if(min!=i)
    	{
			temp = tokFreqVector[i];
			tokFreqVector[i] = tokFreqVector[min];
			tokFreqVector[min] = temp;
    	}
	}
    cout << t << sorted << t  <<endl << "{";
    for(int i = 0; i<tokFreqVector.size()-1; i++)
   {
       cout<<tokFreqVector[i].token<<": "<<tokFreqVector[i].freq<< ", ";
   }
   cout <<"}" << endl;
}
