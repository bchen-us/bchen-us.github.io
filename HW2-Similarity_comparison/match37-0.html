#include <iostream>
#include <vector>
#include <string>

using namespace std;

struct TokenFreq{
	string token;
	int freq;
};

/*
Function to find unique tokens along with their frequencies in a given string
@param string
@return vector<TokenFreq>
*/
vector<TokenFreq> getTokenFreq( string& s){
	
	vector<TokenFreq> tokenFreq;
	
	//return if string is empty
	if(s.empty()){
		return tokenFreq;	
	}	
	
	// appending a white space helps to conveniently add the last element in the vector.
	s.append(" ");
	int pos_first_char = 0;
	
	//Loop to tokenize the string 's' and fill the vector with those tokens 
	for(int idx=1; idx < s.length(); idx++){
		
		if(s[idx] == ' '){
		
			string _token = s.substr(pos_first_char, idx - pos_first_char);
			
			if(tokenFreq.size() == 0){
				
				TokenFreq tf;
				tf.freq = 1;
				tf.token = _token;
				tokenFreq.push_back(tf);
			
			}else{
			
				bool found = false;
			
				// Loop to find if the token already exists in the vector.
				for(int j=0; j < tokenFreq.size(); j++){
					if(tokenFreq[j].token == _token){
						found = true;
						tokenFreq[j].freq += 1;
					}
				}
				if(found == false){
					TokenFreq tf;
					tf.freq = 1;
					tf.token = _token;
					tokenFreq.push_back(tf);
				}
				
			}
			
			pos_first_char = idx+1;
		}
	}
	return tokenFreq;
}

/*
Function to sort the given vector of TokenFreq using Selection Sort. Here parameter is passed as reference.
@param vector<TokenFreq>
@return void
*/
void selectionSort_v1(vector<TokenFreq> &tokFreqVector ){
	
	for(int j=0; j < tokFreqVector.size(); j++){
		
		int minEleIdx = j;
		
		for(int i = j+1; i < tokFreqVector .size(); i++){
			
			if(tokFreqVector[minEleIdx].freq > tokFreqVector[i].freq){
						
				minEleIdx = i;
			}
		}
		
		if( j != minEleIdx){
			TokenFreq temp = tokFreqVector [j];
			tokFreqVector[j] = tokFreqVector[minEleIdx];
			tokFreqVector[minEleIdx] = temp;
		}
	}
	

	for(int i=0; i<tokFreqVector.size(); i++){
		cout<<"("<<tokFreqVector[i].token<<","<<tokFreqVector[i].freq<<") ";
	}
	cout<<"\nSort by passing reference is complete."<<endl;
	cout<<"Address of vector which was passed as reference: "<<&tokFreqVector<<endl;
}

/*
Function to sort the given vector of TokenFreq using Selection Sort. Here parameter is passed as value.
@param vector<TokenFreq>
@return vector<TokenFreq>
*/
vector<TokenFreq> &selectionSort_v2(vector<TokenFreq> tokFreqVector){
	
	for(int j=0; j < tokFreqVector.size(); j++){
		
		int minEleIdx = j;
		
		for(int i = j+1; i < tokFreqVector.size(); i++){
			
			if(tokFreqVector[minEleIdx].freq > tokFreqVector[i].freq){
						
				minEleIdx = i;
			}
		}
		
		TokenFreq temp = tokFreqVector[j];
		tokFreqVector[j] = tokFreqVector[minEleIdx];
		tokFreqVector[minEleIdx] = temp;
		
	}
	return tokFreqVector;
}

int main(){
	
	string s;
	
	cout<<"Enter a text (not more than 100 words):"<<endl;
	getline(cin,s);
	
	cout<<"Tokenizing the text..."<<endl;
	vector<TokenFreq> vec = getTokenFreq(s);
	cout<<"Tokenization complete. Vector size: "<<vec.size()<<endl;
	cout<<"Vector: ";
	for(int i=0; i<vec.size(); i++){
		cout<<"("<<vec[i].token<<","<<vec[i].freq<<") ";
	}
	cout<<"Original Vector address: "<<&vec<<endl;
	
	
	cout<<"\n\nSorting by passing parameter as value..."<<endl;
	vector<TokenFreq> sortByValueVector = selectionSort_v2(vec);
	for(int i=0; i<sortByValueVector.size(); i++){
		cout<<"("<<sortByValueVector[i].token<<","<<sortByValueVector[i].freq<<") ";
	}
	cout<<"\nSort by passing parameter as value is complete."<<endl;
	cout<<"New vector address: "<<&sortByValueVector;
	
	cout<<"\n\nSorting by passing reference as parameter..."<<endl;
	selectionSort_v1(vec);
	
	return 0;
	
}
