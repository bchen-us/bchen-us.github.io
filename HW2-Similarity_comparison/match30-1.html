vector<string> getWords(string sentence)
{
    vector<string> words;
    string temp;
    stringstream ss (sentence);
    while(ss>>temp)
        words.push_back(temp);
    return words;
};

int checkFreq(string s, vector<Tokenfreq> ss)
{

    for(int i=0; i<ss.size(); i++)
    {

        if(s == ss[i].token)
        {
            return i;

        }
    }

    return -1;
};
vector<Tokenfreq> getTokenfreq(string &s){
    vector<string> words;
    vector<Tokenfreq> frequenciis;

    int status;

    Tokenfreq tokenFeq;
    words = getWords(s);
    for(int i=0; i<words.size();i++)
    {

        status=checkFreq(words[i],frequenciis);

        if(status == -1)
        {

            tokenFeq.token = words[i];
            tokenFeq.freq = 1;
            frequenciis.push_back(tokenFeq);

        }
        else
        {
            frequenciis[status].freq ++;



        }
    }


    return frequenciis;

}
