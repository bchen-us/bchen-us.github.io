// C++ program to print the maximum count of the character and print the tokens of string
#include <iostream>
#include <string>
#include <sstream>
using namespace std;

//identify the most frequent letter
char getMostFreqLetter(string s )
{

    int max=0;
    int count=0;
    char maxCharacter=0;

    for(char q=' ';q<='~';q++)
    {

    // filling the character array with count of characters
    count = 0;
    for (int i=0; i<s.length(); i++)
    {
        if(s[i]==q)
                count++;
    }
    if(count>max)
        {
            max=count;
            maxCharacter=q;
        }
    }

    return maxCharacter;

}
//identify and print the tokens contained
void StrToTokens(string s)
{
    // initialize string stream
    istringstream ss(s);
    string token;

    // determine the token
    while(getline(ss, token, ' '))
    {
        // print the token
        cout << token << '\n';
    }
}

// test above two function
int main()
{
    char s[] = "Always remember that you are unique. Just like everyone else." ;
    cout << "Max occurring character is " << getMostFreqLetter(s) << endl;

    cout << "\nString tokens: \n";
    StrToTokens(s);
    cout << endl;
}
