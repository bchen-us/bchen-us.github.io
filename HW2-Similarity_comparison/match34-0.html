char getMostFreqLetter(string s)
{
	int count = 0;
	int highestFreq = 0;
	char mostFreqChar;

    for(char c='A'; c<='z'; c++)     
    {
        count = 0;
        for(int i=0; i<s.length(); i++)
        {
            if(tolower(s[i])==tolower(c))
                count++;
        }
        
        if(count>highestFreq)
        {
            highestFreq = count;
            mostFreqChar = tolower(c);
        }
    }
    cout << "getMostFreqLetter output: Most common letter is '" << mostFreqChar << "' whose frequency is " << highestFreq << "." << endl;
    
    return mostFreqChar;
}

void StrToTokens(string s)
{
	string str = "'";
	for(int i=0; i<s.length(); i++)
        {
            if(s[i]!=' ')
            {
            	str = str + s[i];
			}
			else
			{ 
				if(s[i]!=s[i+1])
				{
				str = str + "'" + " " + "'";
				}
			} 
        }
    str = str + "'";
    cout << "StrToTokens output: " << str;
}
 
int main(int argc, char** argv) 
{
	string s = "Always remember that you are unique.  Just like everyone else.";
	getMostFreqLetter(s);
	StrToTokens(s);
	return 0;
}
