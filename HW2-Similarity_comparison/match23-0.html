vector<TokenFreq> getTokenFreq(string& s)
{
	bool isNotThere;
	vector<TokenFreq> a;
	int lIndex = 0;
	int rIndex = 0;
	for (int i = 0; i < s.length(); i++)
	{
		if ((s[i] == 32))
		{
			rIndex = i;
			TokenFreq current ={s.substr(lIndex, rIndex-lIndex), 1};
			for (int i = 0; i< a.size(); i++)
			{
				if (current.token == a[i].token)
				{
					a[i].freq += 1;
					isNotThere = false;
				} 
			}
			if (isNotThere)
			{
				a.push_back(current);
			}
			lIndex = rIndex +1;
		}
	isNotThere = true;
	}
	TokenFreq current ={s.substr(lIndex, s.length()-rIndex-1), 1};
	for (int i = 0; i< a.size(); i++)
	{
		if (current.token == a[i].token)
		{
			a[i].freq += 1;
			isNotThere = false;
		} 
	}
	if (isNotThere)
	{
		a.push_back(current);
	}
	return a;
}
void Swap (int a, int b, vector<TokenFreq>& tokFreqVector){
		TokenFreq temp = tokFreqVector[a];
		tokFreqVector[a] = tokFreqVector[b];
		tokFreqVector[b] = temp;
}
void selectionSort_v1(vector<TokenFreq>& tokFreqVector)
{
	for(int i = 0; i < tokFreqVector.size()-1; i++)
	{
		int small = i;
		for(int j = i + 1; j< tokFreqVector.size(); j++)
		{
			if (tokFreqVector[small].freq > tokFreqVector[j].freq)
			{
				small = j;
			}
		}
		Swap(small, i, tokFreqVector);
	} 
}
vector<TokenFreq>& selectionSort_v2(vector<TokenFreq> tokFreqVector)
{
	for(int i = 0; i < tokFreqVector.size()-1; i++)
	{
		int small = i;
		for(int j = i + 1; j< tokFreqVector.size(); j++)
		{
			if (tokFreqVector[small].freq > tokFreqVector[j].freq)
			{
				small = j;
			}
		}
		Swap(small, i, tokFreqVector);
	}
	return tokFreqVector;
	
}
