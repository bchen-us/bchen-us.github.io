#include <iostream>
#include <vector>

using namespace std;

struct TokenFreq {
    string token;
    int freq;
};

vector<TokenFreq> getTokenFreq(string &s) {

    vector<TokenFreq> returnVector;

    {
        string word = "";
        int index = 0;
        int f = 0;
        int flag = 0;

        returnVector.push_back(TokenFreq());//puts each word into the vector

        for (int i = 0; i < s.length(); i++) {
            if (s[i] == ' ') {
                flag = 0;
                for (int k = 0; k < returnVector.size(); k++) {
                    if (returnVector[k].token == word) {
                        returnVector[k].freq++;
                        flag = 1;
                        word = "";
                    }
                }
                if (flag == 0) {
                    f++;
                    returnVector[index].token = word;
                    returnVector[index].freq = 1;
                    index++;
                    returnVector.push_back(TokenFreq());
                    word = "";
                }
            }
            else {
                word = word + s[i];
            }
        }
        flag = 0;
        for (int k = 0; k < returnVector.size(); k++) {
            if (returnVector[k].token == word) {
                returnVector[k].freq++;
                flag = 1;
                word = "";
            }
        }
        if (flag == 0) {
            cout << word;
            returnVector[index].token = word;
            returnVector[index].freq = 1;
        }
        cout << "frequency of each word:" << endl;

        for (int k = 0; k < returnVector.size() - 1; k++) {
            cout << returnVector[k].token << ": " << returnVector[k].freq << endl;
        }
        return returnVector;
    }
}


void selectionSort_v1( vector<TokenFreq> & tokFreqVector )
{
    TokenFreq temp;
    int sort;

    for( int i = 0; i < tokFreqVector.size()-2; i++){
        sort = i;
        for (int j = i + 1; j <tokFreqVector.size()-1; j++){
            if (tokFreqVector[j].freq < tokFreqVector[sort].freq)
                sort = j;
        }
        if(sort != i)
        {
            temp = tokFreqVector[i];
            tokFreqVector[i] = tokFreqVector[sort];
            tokFreqVector[sort] = temp;
        }
    }
    cout<<"sorted frequency of words:"<<endl;

    for(int i = 0; i<tokFreqVector.size() - 1; i++)
    {

        cout<<tokFreqVector[i].token<<": "<<tokFreqVector[i].freq<<endl;
    }
}
