//declaring the struct TokenFreq with string and frequency
struct TokenFreq
{
    string token;
    int freq;
};
/*
 This function recieves a vector of TokenFreq objects by refrence and value,
 then sort in increasing order of the token frequency
 */

void selectionSort_v1(vector<TokenFreq> & tokFreqVector )
{
    TokenFreq temp;
    int minim;
    
    for(int i = 0; i <tokFreqVector.size()-1; i++){
        minim = i;
        for (int j = i+1; j <tokFreqVector.size()-1; j++){
            if (tokFreqVector[j].freq < tokFreqVector[minim].freq)
                minim = j;
        }
        if(minim!=i)
        {
            temp = tokFreqVector[i];
            tokFreqVector[i] = tokFreqVector[minim];
            tokFreqVector[minim] = temp;
        }
    }
    cout<<"\nTokens and Frequency with increasing order:"<<endl;
    for(int i = 0; i<tokFreqVector.size()-1; i++)
    {
        
        cout<<tokFreqVector[i].token<<": "<<tokFreqVector[i].freq<<endl;
    }

}
