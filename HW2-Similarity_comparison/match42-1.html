#include <string>
#include <iostream>
#include <sstream>
#include <vector>

using namespace std;

struct TokenFreq { //struct with two variables for a token and its frequency
	string token;
	int freq;
};

vector<TokenFreq> getTokenFreq(string & s) {
	vector<TokenFreq> pairedTokens;	//a vector of structs
	istringstream ss(s);
	string temp;
	TokenFreq s1;	//new struct
	int count = 0;

	while (ss >> temp) {
		s1.token = temp;	//assigns token from stream to TokenFreq variable token
		s1.freq = count;
		for (int i = 0; i < s.size() - 1; i++) {  //loop through inputted string
			if (pairedTokens.at(i).token == s1.token) {  //check to see if token at a location in the vector is the same as token from the stream
				count++;
				s1.freq = count;
			}
			else {		//if it does not already exist in the vector add it and set the count to 1
				count = 1;

				s1.freq = count;
			}
			pairedTokens.push_back(s1);  //Push TokenFreq object into the TokenFreq vector
		}
		cout << s1.token << " " << s1.freq << endl; //prints token to console
	}

	return pairedTokens;
}

void selectionSort_v1(vector<TokenFreq>& tokFreqVector) { //selection sort by reference
	for (int j = 0; j < tokFreqVector.size() - 1; j++) {
		int min = j;
		for (int i = j + 1; i < tokFreqVector.size(); i++) {
			if (tokFreqVector.at(i).freq < tokFreqVector.at(min).freq) {	//if frequency at 1st position is less than the one at 0 position, switch them
				min = tokFreqVector.at(i).freq;
			}
		}

		if (min != j) {
			swap(tokFreqVector.at(j), tokFreqVector.at(min));
		}
	}
}

vector<TokenFreq> selectionSort_v2(vector<TokenFreq> tokFreqVector) { //selection sort by value

	for (int j = 0; j < tokFreqVector.size() - 1; j++) {
		int min = j;
		for (int i = j + 1; i < tokFreqVector.size(); i++) {
			if (tokFreqVector.at(i).freq < tokFreqVector.at(min).freq) {
				min = tokFreqVector.at(i).freq;
			}
		}

		if (min != j) {
			swap(tokFreqVector.at(j), tokFreqVector.at(min));
		}
	}

	return tokFreqVector;
}

int main() {
	string s;
	/*string test;

	getline(cin,s);

	cout << "Token/Frequency pairs: " << endl;
	vector<TokenFreq> a1 = getTokenFreq(s);

	cout << "Selection Sort Version 1: " << endl;
	selectionSort_v1(a1);

	cout << "Selection Sort Version 2: " << endl;
	selectionSort_v2(a1);

	return 0;
}
