vector<TokenFreq> getTokenFreq( string& s) {
    TokenFreq temp;
    vector<TokenFreq> tok; //vector that uses the struct
    vector<string> elems; //extra vector result from splitting the string
    stringstream stream;    //rips out all of the words of the string
    stream.str(s); // links the string s to stringstream
    string item;
    while(getline(stream, item,' ')){
        if(!item.empty()) {   //if the string is not empty and has a word then another is added
            elems.push_back(item);

        }
    }
    map< string , int > map1; // assigns each key thats a string an integer value

    for(int i=0; i<=elems.size(); i++){

        map1[elems[i]]+=1; //adds 1 to the index of each word

    }
    for( map<string,int>::iterator it = map1.begin(); it != map1.end();it++){ //goes through the entire map

        cout << it->first << " = " << it->second << endl;
        temp.token = it->first; //makes token equal to the x value in the pair (x,y)
        temp.freq = it->second; //makes frequency equal to the y value in the pair (x,y)
        tok.push_back(temp); //adds a new element in the end of the vector
    }
    //puts the contents from the string vector into the struct vector
    return tok;
    }

void selectionSort_v1( vector<TokenFreq> & tokFreqVector ){
// this function is where the sort is modified.
    int x, y;
    for (x = 0; x < tokFreqVector.size() - 1; x++) { //will go through the vector of tokens
        int min = x; //minimum is used to store the index with smallest value

        for (y = x + 1; y < tokFreqVector.size(); y++) { //y will be compared with x to see which is bigger
            if (tokFreqVector[y].freq < tokFreqVector[min].freq) { //if frequency of the second variable is smaller than
                // the min, which was the x value than y is the new min
                min = y;
            }
        }

        swap(x, min, tokFreqVector); // if the index of y is smaller it will be swapped with x
    }
}

void swap(int a, int b, vector<TokenFreq>& tokFreqVector){
    TokenFreq temp =  tokFreqVector[a]; // a function to swap two values
    // that will be used in the selectionSort_v1 function
    tokFreqVector[a]=tokFreqVector[b];
    tokFreqVector[b]= temp;
}

vector<TokenFreq>& selectionSort_v2(vector<TokenFreq>& tokFreqVector ) {
    vector<TokenFreq> tok2 = tokFreqVector; //making a copy of the vector to this function

    int x, y; //does same as selectionSort_v1, except this one returns the value from the function
    for (x = 0; x < tok2.size() - 1; x++) {
        int min = x;

        for (y = x + 1; y < tok2.size(); y++) {
            if (tok2[y].freq < tok2[min].freq) {
                min = y;
            }
        }
        swap(x, min, tok2);
        return tok2;
    }
}
