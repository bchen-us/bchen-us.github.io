#include <string>

struct TokenFreq
{
public:
    string token;
    int freq;
};

vector<TokenFreq> getTokenFreq(string& s);
void selectionSort1(vector<TokenFreq>& tokenFreqVector);
vector <TokenFreq>& selectionSort2(vector<TokenFreq> tokenFreqVector);
void swap (int x, int y, vector<TokenFreq>& tokFreqVector); // swap switches x and y

vector<TokenFreq> getTokenFreq(string& s) // creating a vector of token frequencies

{
    vector<TokenFreq> one; //
    int left =0;
    int right=0;
    bool notConfirmed;
    for( int i=0; i< s.length(); i++)
    {
        if(s[i] ==32)
            
        {
            right=i;
            TokenFreq current ={ s.substr(left, right-left), 1}; // creating a substring to read the size.makes the size into a token then puts it into the variable
            for(int i=0; i< one.size(); i++)
            {
                if ( current.token == one[i].token)
                {
                    one[i].freq +=1;
                    notConfirmed =false;
                }
            }
            if (notConfirmed)
            {
                one.push_back(current);
            }
            left =right +1;
            
        }
        notConfirmed = true;
    }
    TokenFreq current ={s.substr(left, s.length()-right-1), 1};
    for (int i=0; i<one.size(); i++)
    {
        if(current.token ==one[i].token)
        {
            one[i].freq +=1;
            notConfirmed= false;
        }
    }
    if(notConfirmed)
    {
        one.push_back(current);
    }
    return one;
    
}
void swap (int x, int y, vector<TokenFreq>& tokFreqVector){ //manually put a temporary variable where x goes into temporary y goes into x then x moves to y
    TokenFreq Temporary =tokFreqVector [x];
    tokFreqVector[x] = tokFreqVector[y];
    tokFreqVector[y] = Temporary;
    
}
void selectionSort1(vector<TokenFreq>& tokFreqVector) // takes first element in the vector compares to the second and if its less it swaps and if its more it keeps checking
{
    for(int i=0; i< tokFreqVector.size()-1; i++)
    {
        int less =i;
        for(int j = i+1; j< tokFreqVector.size(); j++)
        {
            if(tokFreqVector[less].freq > tokFreqVector[j].freq)
            {
                less = j;
            }
        }
        swap(less, i, tokFreqVector);
        
        
    }
}
vector<TokenFreq>& selectionSort2(vector<TokenFreq> tokFreqVector) // this function returns a sorted vector
{
    for(int i=0; i< tokFreqVector.size()-1; i++)
    {
        int less =i;
        for(int j=i +1; j<tokFreqVector.size(); j++)
        {
            if(tokFreqVector[less].freq > tokFreqVector[j].freq)
            {
                less =j;
            }
        }
        swap(less, i, tokFreqVector);
    }
    return tokFreqVector;
}
