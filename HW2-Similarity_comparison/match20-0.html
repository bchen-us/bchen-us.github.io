#include<iostream>
#include<string>
#include<vector>

using namespace std;
struct TokenFreq {
    string token;
    int freq;
};

/* function identifies all unique tokens in the string s, and its frequency
 * after identification, the tokens are stored in a vector paired as (token, freq)*/

vector<TokenFreq> getTokenFreq(string &s) {

    int index = 0;
    vector<TokenFreq> TokFreq; //vector that stores all the vector pairs (token, freq)
    string word = "";
    TokFreq.push_back(TokenFreq());
    int f = 0;
    int indicate = 0;

    // this for loop will check for the unique tokens and add to TokFreq in pairs (token, freq)
    for (int j = 0; j < s.length(); j++) {
        if (s[j] == ' ' || s[j] == '.' || s[j] == ',' || s[j] == ';' || s[j] == '!' || s[j] == ':' || s[j] == '?') {
            indicate = 0;
            for (int i = 0; i < TokFreq.size(); i++) {
                if (TokFreq[i].token == word) {

                    TokFreq[i].freq++;
                    indicate = 1;
                    word = "";
                }


            }

            if (indicate == 0) {
                f++;
                TokFreq[index].token = word;
                TokFreq[index].freq = 1;
                index++;
                TokFreq.push_back(TokenFreq());

                word = "";
            }

        } else {
            word = word + s[j];

        }
    }

    indicate = 0;
    for (int i = 0; i < TokFreq.size(); i++) {
        if (TokFreq[i].token == word) {
            TokFreq[i].freq++;
            indicate = 1;
            word = "";
        }


    }
    if (indicate == 0) {
        cout << word;
        TokFreq[index].token = word;
        TokFreq[index].freq = 1;
        index++;
    }


    cout
            << "\nAfter having called the getTokenFreq() function, you should identify the following list of (token, freq) pairs and store them in a vector"
            << endl;

    for (int i = 0; i < TokFreq.size() - 1; i++) {

        cout << TokFreq[i].token << ", " << TokFreq[i].freq << endl;
    }

    return TokFreq;
}

//receives a vector of TokenFreq by reference, and applies to sort algorithm
void selectionSort_v1(vector<TokenFreq> &tokFreqVector) {
    TokenFreq temp;
    int minimum;

    for (int i = 0; i < tokFreqVector.size() - 2; i++) {
        minimum = i;
        for (int j = i + 1; j < tokFreqVector.size() - 1; j++) {
            if (tokFreqVector[j].freq < tokFreqVector[minimum].freq)
                minimum = j;
        }
        if (minimum != i) {
            temp = tokFreqVector[i];
            tokFreqVector[i] = tokFreqVector[minimum];
            tokFreqVector[minimum] = temp;
        }
    }
    cout << " \nAfter having called either selection sort v1, or v2, the sorted vector of token-freq pairs"
            " will contain the following information: " << endl;
    for (int i = 0; i < tokFreqVector.size() - 1; i++) {

        cout << tokFreqVector[i].token << ", " << tokFreqVector[i].freq << endl;
    }
}
