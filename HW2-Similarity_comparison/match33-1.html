// C++ program to identify most frequent letter and 
// the tokens contained in s on the standard output

#include <iostream>
#include <string>
#include <sstream>
using namespace std;

char getMostFreqLetter(string  s)
{
    
    int max=0;
    int count=0;
    char maxCharacter=0;

    for(char q=' ';q<='~';q++)
    {
    
    // function to identify the most frequent letter
    count = 0;
    for (int i=0; i<s.length(); i++)
    {
        if(s[i]==q)
                count++;
    }
    if(count>max)
        {
            max=count;
            maxCharacter=q;
        }
    }
    
    return maxCharacter;
        
}
    
void StrToTokens(string s)
{
    // initializes string stream
    istringstream ss(s);
    string token;
    
    // implementation of function to identify tokens
    while(getline(ss, token, ' '))
    {
        // prints the token
        cout << token << '\n';
    }
}

//  the main function that declares and initializes 
// the string s, and  calls the above two functions. 

int main()
{
    char s[] = "In every walk with nature one receives far more he seeks.";
    cout << "Max occurring character is: " << getMostFreqLetter(s) << endl;
    
    cout << "\nString tokens: \n";
    StrToTokens(s);
    cout << endl;
}
