#include <iostream>
#include<vector>

using namespace std;
//*String to Token*//
void StrToTokens(string s)
{
    string tokens[100];
    int WordCount = 0;
    vector <char> word;
    ///iterate through string
    for(int i=0; i<=s.length(); i++)
    {
        if(s[i] != ' ')//skip spaces
        {
            if(s[i] != '\0')//skip end of string
                
            {
                word.push_back(s[i]);///copy char to word
            }
        }
        else
        {
            if(i>0) //ignore empty first char
            {
                if(s[i-1]!= ' ')
                    //if the previous letter is also not a space then it is the end of a token.
                {
                    word.clear(); WordCount++;
                   
                }
            }
        }
        
        
        ///copy token(one char at a time) to string array
        string token(word.begin(),word.end());
        tokens[WordCount]=token;
    }
    ///print tokens
    
    
    for (int out=0;out<=WordCount; out++)
    {
        cout << tokens[out]<<endl;
    }
    
    
    
}


//*Letter Frequency*//
void getLetterFreq(string s)
{
    cout  <<"Your paragraph contains "<<endl;
    for(char alpha = 'A'; alpha <= 'Z'; alpha++)    ///iter uppercase alphabet
    {
        int count = 0;
        for(int i = 0; i <= s.length(); i++)
        {
            if(toupper(s[i])==alpha) /// toupper counts lowercase==uppercase
            {
                count++;
            }
            
        }
        if(count>0)//print each alphabet count
        {
            cout << count << " " << alpha << "'s" << endl;
        }
        
    }
}

//*Main*//
int main(int argc, const char * argv[]) {
    
    string s = "Always remember that you are unique.  Just like everyone else.";
    
    cout << "Enter a paragraph:";
    getline(cin, s);

    cout<< " Letter Frequency:"<<endl;
    getLetterFreq(s);
    
    cout<<" Words in the string:"<<endl;
    StrToTokens(s);
    
    
    return 0;
}
