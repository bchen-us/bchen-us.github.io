#include <iostream>
#include <vector>

using namespace std;

// Function for finding out the frequency of letters that show up in given sting
void getLetterFreq(string s){
   // Initalize char array that lists letters to look for, a-z
   char alphabet[] = "abcdefghijklmnopqrstuvwxyz";

   // For loop that checks for each letter and stores results in an array
   for(int i = 0; i < 26; i++){
      char letter = alphabet[i];

      int counter = 0;
      for(int j = 0; j < s.length(); j++){
         if(s[j] == letter){
            counter++;
         }
      }
      // If loop that prints only if the letter shows up at least once

      if(counter != 0){
         cout<<letter<<": "<< counter << endl;
      }
   }
}

// Takes string s and splits words into tokens then stored in a vector
void StrToTokens(string s){
   // Initialize vector thats gonna store tokens called list
   vector<string> list;
   string word = "";
   s = s + " ";

   // For loop that parses through string and pushes "word" when it passes
   // a whitespace
   for(int i = 0; i < s.length(); i++){
      if(s[i] == ' '){
         // Makes sure that "word" being pushed through isn't whitespace
         if(word != ""){
            list.push_back(word);
         }
         word = "";
      }
      else{
         word = word + s[i];
      }
   }

   // For loop that prints out each element in vector list
   for(int i = 0; i < list.size(); i++){
      if(i == (list.size()-1)){
         cout<< list[i] << endl;
      }
      else
         cout << list[i] << ",";
   }

}

// Main function that initalizes s by prompting user for input from console
// Then passes it through functions created
int main()
{
   string s;
   cout << "Enter your paragraph:" << endl;
   getline(cin,s);
   cout << endl << endl;
   cout << "Letter Count: " << endl;
   getLetterFreq(s);
   cout << endl << endl;
   cout << "Tokens: ";
   StrToTokens(s);
   cout << endl << endl;
}
