#include <vector>
#include <iostream>
#include <string>

using namespace std;

void getLetterFreq(string s){
    
    
    vector<int> frequency (26);
    
    
    char alpha[] = "ABCDEFGHIJKLMNOPQRSTUVWXYZ" ;
    vector<char> alphabet( alpha, alpha+sizeof(alpha)-1 ) ;
    
    //converts string to upper case
    for (int i=0; i<s.length(); ++i)
        s[i]=toupper(s[i]);

    //parse for frequencies

    for(int parse = 0; parse < s.size(); parse++){
        for(int j = 0; j < alphabet.size(); j++){
            if (s.at(parse) == alphabet[j]){
                frequency[j]++;
            }
                
        }
    }
    
    //printout of frequencies
    for(int i = 0; i < alphabet.size(); i++){
        cout << alphabet[i] << ": " << frequency[i] << endl;
    }
    cout << endl;
}

void StrToTokens(string s){
    
    
    vector<char> words;
    
    //formatting
    cout << "'";
    
    for(int i = 0; i < s.size(); i++){
        
        //pushback valid chars to vector
        if (s.at(i) != ' '){
            words.push_back(s.at(i));
            cout << s.at(i);
        }
        
        //print vector and clear
        if (s.at(i) == ' '){
            cout << "', '";
            words.clear();
            
        }
    }
    cout << "'" << endl << endl;
}


int main(int argc, const char * argv[]) {
    // insert code here...
    string s ="In a hole in the ground there lived a hobbit. Not a nasty, dirty, wet hole, filled with the ends of worms and an oozy smell, nor yet a dry, bare, sandy hole with nothing in it to sit down on or to eat: it was a hobbit-hole, and that means comfort.";
    
    getLetterFreq(s);
    
    StrToTokens(s);
    
    return 0;
}
