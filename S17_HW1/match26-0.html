#include <iostream>
#include <string>
using namespace std;

//declare empty functions with string parameter
void getLetterFreq(string);
void StrToTokens(string );

//receive input from user and call functions
int main () {
	string s;
	cout << "Enter a sentence/paragraph (Enter all information at once, word limit=100) : " << endl;
	getline(cin, s);
	
	//check to see if the input is less than 100 words 
	if (s.length() > 100) {
		cout << "You had more than 100 words, please try again." << endl;
	}

	 	getLetterFreq(s);
		StrToTokens(s);
}

//function to get the frequency of each letter from the input
void getLetterFreq(string s) {
	int alphaFreq [26];
	char alphaArray [26] = {'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l', 'm', 'n', 'o', 'p', 'q', 'r', 's', 't', 'u', 'v', 'w', 'x', 'y', 'z'};
	
	//loop to go through 26 spaces in alphaArray
	for (int a = 0 ; a < 26; a++) {
		int counter = 0;
		
		//loop to go through every space in the input 
		for (int b = 0; b < s.length(); b++) {
			
			//check if space in input consists of character provided in alphaArray
			if (s[b] == alphaArray[a]) { 
				counter++;	
			}
		}
		alphaFreq [a] = counter;
	}
	cout << "The letter frequencies are: " << endl;
	
	//loop to print letter frequencies 
	for (int c = 0; c < 26; c++) {
		if (alphaFreq [c] != 0) {
			cout << alphaArray[c] << ": " << alphaFreq[c] << " ";
		}
	}
	cout << endl;
}

//function to identify and print the tokens from the input
void StrToTokens(string s) {
   	string token = "";
   	string savedToken ="";
	cout << "The tokens will be: ";
	
	//loop going through string to check for delimiters
   	for (int i = 0; i <= s.size(); i++) {
       	if (s[i] == ' ') {
       		
       		//if there is not delimiter print the token
            if (token.size() != 0) {
            	cout << "'" << token << "', ";
				token = "";
            }
        
		//move to next space if there is a delimiter
        } else {
           	token = token + s[i];  
       	}
   	}
    if (s.size() != 0) { 
		cout << "'" << token << "'. " << endl;
	}
	savedToken = token;
}
