#include <iostream>
#include <string>

using namespace std;

void StrToTokens(string);
void getLetterFreq(string);

int main() {
    string name;
    cout << " Please enter your name: ";
    getline(cin, name);
    cout << name << "\n";
    if (name.length() > 100) {
        cout << " Your string holds more than 100 tokens. Please enter a valid string, consisting of no more than 100 tokens. " << endl;
    }

    getLetterFreq(name);
    StrToTokens(name);
}

void getLetterFreq(string input) {
    int frquency[26];
    char letter[26] = {'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l', 'm', 'n', 'o', 'p', 'q', 'r', 's', 't', 'u', 'v', 'w', 'x', 'y', 'z'};
    for (int i = 0; i < 26; i++) {
        int count = 0;
        for (int j = 0; j < input.length(); j++) {
            if (input[j] == letter[i]) {
                count++;
            }
        }
        frquency[i] = count;
    }
    for (int k = 0; k < 26; k++) {
        cout << "Letter " << letter[k] << " has frequency : " << frquency[k] << endl;
    }
}

void StrToTokens(string input) {
    int startindex = 0;
    int reInitializePointer = 0;
    int j = 0;
    do {
        char strchar[100];
        for (int i = startindex; i < input.length(); i++) {
            if (input[i] != ' ') {
                startindex++;
                strchar[j] = input[i];
                j++;
            } else {
                startindex = startindex + 1;
                reInitializePointer = j;
                j = 0;
                break;
            }
        }
        cout << strchar << endl;
        // Re-initializing the array so that it doesn't hold its previous array values.
        for (int k = 0; k < reInitializePointer; k++) {
            strchar[k] = ' ';
        }
    } while (startindex < input.length());
}
