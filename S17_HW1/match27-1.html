#include <iostream>
#include <vector>
#include <string>
#include <map>
#include <algorithm>


using namespace std;


//converts all tolower and removes spaces from the statement given
void getLetterFreq(string s) {
    transform(s.begin(), s.end(), s.begin(), ::tolower);
    s.erase(std::remove(s.begin(), s.end(), ' '), s.end());
    map<char,int> chars;
    for (unsigned int j=0; j<s.size(); j++)
        chars[s[j]]++;
    for (map<char, int>::iterator iter = chars.begin(); iter != chars.end(); iter++)
        cout << iter->first << ": " << iter->second << ", ";
}


void StrToTokens(string s){
    std::vector<string> arr;
    string fill = "";

    for (unsigned int j=0; j<s.size(); j++)
    {
        if (!isspace(s[j]))
        {
            fill = fill + s[j];
        }
        if (isspace(s[j]))
        {
            if(fill!=""){
                arr.push_back(fill);
            }
            
            fill = "";
        }
    }
    arr.push_back(fill);
    for(int i=0; i<arr.size(); ++i)
    {std::cout << arr[i] +", ";}
    cout <<"\n";
    
}


int main()
{
    string s ="My name is Devesh Patel ";
    getLetterFreq(s);
    StrToTokens(s);
}
