#include <iostream>

void getLetterFreq(string input);

int main()
{

void getLetterFreq(string input)
{
    vector<char> charsInString; // creating a vector to hold the characters that are in the string
    for (int i = 0; i < input.size(); i++)
    {
        if (charsInString.empty() && input[i] != ' ') // if the vector is empty, insert the first charater from the string in to the vector
        {
            charsInString.push_back(tolower(input[i]));
        }
        
        else if (find(charsInString.begin(), charsInString.end(), tolower(input[i])) == charsInString.end() && input[i] != ' ') // checking if the next character from the string is in the vector
        {
            charsInString.push_back(tolower(input[i]));
        }
    }
    
    int charCount = 0; // creating a counter to count the occurence of the letters
    
    cout << "Frequency of the letters" << endl;
    for (int j = 0; j < charsInString.size(); j++)
    {
        for (int l = 0; l < input.size(); l++)
        {
            if (charsInString[j] == tolower(input[l]))
            {
                charCount++;
            }
        }
        cout << charsInString.at(j) << ": " << charCount << endl;
        charCount = 0;
    }
}

void StrToTokens(string input)
{
    string userToken;
    vector<string> separatedTokens;

    for (int i = 0; i < input.size(); i++)
    {
        if(input[i] != ' ' && input[i] != '\t') // iterating through each char of the string and checking if its a space or \t
        {
            userToken += input[i]; // if its not a space or \t, then the char is added to a string
        }
        else
        {
            separatedTokens.push_back(userToken); // if it hits a space or \t, it adds the string in to the vector
            userToken = ""; // clears the string to be resued
        }
    }
    
    if (userToken.length() > 0) // this is here because the message may not end with a delimiter
    {
        separatedTokens.push_back(userToken);
    }
    
    if (separatedTokens.size() > 100)
    {
        cout << "Sorry, there are greater than 100 tokens inputted." << endl;
        return;
    }
    
    vector<string>::iterator i;
    cout << "The tokens of the string are: " << endl;
    for (int j = 0; j < separatedTokens.size(); j++)
    {
        cout << "'" << separatedTokens.at(j) << "'" << ", ";
    }
    cout << endl;
}
