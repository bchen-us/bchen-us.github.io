void getLetterFreq(string s){

    char alphabetArray[26] = { 'a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z'};
    //create an array of all the letters in the alphabet

    int frequencyArray[26];
    //create an array that counts the frequency of the letter at the same position of the letter


    for(int index = 0 ; index < 26 ; index++){
        //this is a for loop to go through and count the frequency of each letter, that's why index is less than 26 like the alphabet array
        int countLetter = 0;
        // start the count at zero for each letter

        for(int position = 0 ; position < s.length(); position++){
            if(s[position] == alphabetArray[index]){
                countLetter++;
                // the count will grow as long as the letter in string s is the same as in the alphabet index position
            }
        }

        frequencyArray[index] = countLetter;
        // the frequency of the letter at spot index will be whatever count is
    }

    for (int positionTwo = 0 ; positionTwo < 26 ; positionTwo++ ) {
        //positionTwo starts at 0 and goes 27 times to go through each letter
        cout << alphabetArray[positionTwo] << "=" << frequencyArray[positionTwo] << ", " ;
        // this statement prints the letter at z location in the alphabetArray and the frequency at positionTwo location in the frequencyArray
    }
    // the above for loop is used to print out the letter with its corresponding frequency within the string

    cout << "\n" ;
    // prints a new line
}

void StrToTokens(string s) {

    for (int index = 0; s[index] != '\0'; index++) {
        //The above for loop makes sure that the position index is in inside string s is an alphabetic letter, creating a token

        printf("%c", s[index]);
        // using %c to print the character at the location in s at position index

        if (s[index] == ' ') {
            printf("\n");
        }
        // the above if statement says that if the position index is in in string s is a non alphabetic letter, it will go to a new line to start the next token
    }
}
