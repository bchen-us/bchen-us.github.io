void getLetterFreq(string input){
char ascii;
int frquency[52];
char alphabet[52] = { 'A','a','B','b','C','c','D','d','E','e','F','f','G','g','H',
					'h','I','i','J','j','K','k','L','l','M','m','N','n','O','o','P','p',
					'Q','q','R','r','S','s','T','t','U','u','V','v','W','w','X','x','Y','y','Z','z'};

for(int i = 0 ; i < 52 ; i++){
int count = 0;
for(int j = 0 ; j < input.length(); j++){
if(input[j] == alphabet[i]){
count++;   
}

}
frquency[i] = count;
}
for(int k = 0 ; k < 52 ; k++ ){
cout << alphabet[k] << " has a frequency of : " << frquency[k] << endl ;
}
}
