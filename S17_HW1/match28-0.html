#include <iostream>

void getLetterFreq(string s){
	int n=s.size();
	int freq[26]={0};
	char letters[26]={'a','b','c','d','e','f','g','h','i','j','k','l',
				      'm','n','o','p','q','r','s','t','u','v','w','x','y','z'};
	for(int i= 0; i<= n ; i++){
	for(int j=0; j<=26 ;j++){
		if (s[i]==letters[j]){
		freq[j]++;
	}
}
}
for (int k=0; k<=26; k++){
	cout <<"This paragraph "<< letters[k]<<" has frequency: "<<freq[k]<<endl;

}
}

       if(s[i]==' ') {
               cout<< t <<endl;
               t="";
               }
