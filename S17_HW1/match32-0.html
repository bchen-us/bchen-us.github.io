#include <iostream>
#include <vector>

using namespace std;

// function to get the frequency & print them out.
void getLetterFreq(string s){
    int freqAZ[26];
    char aZ[26] = {'a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z'};
    
    //initial freqAZ.
    for (int i = 0; i < 26; i++){
        freqAZ[i] = 0;
    }
    
    //count the frequency.
    for(int i = 0; i < s.length(); i++){
        freqAZ[s[i] - 'a']++;
    }
    
    for(int i = 0; i < 26; i++){
        if (freqAZ[i] != 0){
            cout<<aZ[i]<<": "<<freqAZ[i]<<endl;
        }
    }
    
}


// function to get the token & print them out.
void StrToTokens(string s){
    vector<string> list;
    int begin = 0;
    char edlimiter = ' ';
    int cur = 0;
    
    
    // igore the edlimiter in the begin.
    while(s[cur] == edlimiter){
        cur = cur + 1;
        begin = cur;
    }
    
    
    //get the token & put them in the list
    for (int i = cur; i<= s.length(); i++){
        if (s[i] == edlimiter || s[i] == 0){
            if (s[i - 1] != edlimiter){
                string token = s.substr(begin, i-begin);
                list.push_back(token);
                begin = i + 1;
            }
            begin = i + 1;
        }
    }
    
    
    
    //print out those token
    for(int i = 0; i < list.size(); i++){
        cout<<"'"<<list[i]<<"'"<<", ";
    }
    cout<<endl;
    
}



int main() {
    string s;
    cout<<"Enter a string:\n ";
    getline(cin, s);
    
    
    //set string to lower case.
    for (int i = 0; i < s.length(); i++){
        s[i] = tolower(s[i]);
    }
    
    //get the frequency & print them out.
    getLetterFreq(s);
    
    //get the tokens & print them out.
    StrToTokens(s);
    
    return 0;
}
