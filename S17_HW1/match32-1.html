void getLetterFreq(string s)
{
	int letterCount[25];

	{
		if(s[i] >= 65 && s[i] <= 90)	//Checks for uppercase letters

			s[i] = s[i] + 32;	//Changes uppercase to lowercase
		}

void StrToTokens(string s)
{
	int tokenCount= 0;
	int vectorPlacement = 0;
	string token;
	vector<string> tokens;

	for(int i = 0; i <= s.length(); i++)	//Runs through the paragraph.
	{
		if(s[i] != ' ' && i != s.length())	//Checks for whitespace or end of paragraph
		{
			token.push_back(s[i]);			//Places the letters of the word into a string
		}
		else
		{
			tokens.push_back(token);		//Places the word into the vector
			token.clear();					//Clears the word

			vectorPlacement++;
		}
	}
	for(int j = 0; j <= tokens.size() - 1; j++)
	{
		cout<<tokens[j]<< " ";
	}
}
