#include <iostream>
#include <string>
#include <vector>


using namespace std;


void getLetterFreq(string s){
    
    //initialize an array to store the frequency of each letter in the alphabet
    int freq[26];
    
    //initialize a character array of size 26 letters in the alphabet
    char letters[26] = { 'a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z'};
    
    //Loop i: loop thru and count the number of frequency of the 26 letters
    for(int i = 0 ; i < 26 ; i++){
        
        //Loop j: loop thru the 26 letters
        int count = 0;
        for(int j = 0 ; j < 26; j++){
            
            //set the condition that the j's for-loop only loop until it reaches z (letter[25]) or else count will terminate
            if(s[j] == letters[i]){
                count++;
            } // end if
        } //end j's for-loop
        
        freq[i] = count;
    } //end i's for-loop
    
    //Loop k: loop thru and display both the 26 letter and their frequencies onto the show screen
    for(int k = 0 ; k < 26 ; k++ ){
        cout << letters[k] << " has frequency of : " << freq[k] << endl ;
    } // end k's loop
} //end getLetterFreq()


void StrToTokens(string s){
    
    cout << "The Tokens are: " << "\t" << endl;
    
    //Initialize a vector to store the string of input word into the vector @var array
    vector<string> array;
    
    //Initialize an empty string of @var str
    string str = " ";
    
    //Loop to check for any white space found in the input string
    for(int i=0; i<s.size(); i++){
        
        //Set condition using isspace to check for white space
        //If there is NO white space, then the first token is added to the vector
        if (!isspace(s[i])){
            str += s[i];
        } //end if
        //If there is a white space between two tokens, then use push_back() to add new token at the end of the vector
        if (isspace(s[i])){
        
            if(str!= ""){
                array.push_back(str);
            } //end if
            
        str = "";
        } //end if
    } // end for-loop
    
    
    //Make a for-loop to display all the tokens in the vector
    array.push_back(str);
    for(int i=0; i<array.size(); i++){
        cout << array[i] + ",  ";
    } //end for-loop
    cout << endl;
    
    
}



int main(){

    string s;
    cout << "Type in your text here:  \n";
    getline(cin,s);
    
    //Set the condition that paragraph consists no more than 100 tokens
    if(s.length() > 100){
        cout << "Your text contains more than 100 tokens. Invalid input!" << endl;
    }else{
        cout << s <<  "\n";
    }//end if-else

    //Call the above 2 functions
    getLetterFreq(s);
    StrToTokens(s);
    
    return 0;
    
    }
