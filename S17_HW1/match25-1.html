#include <iostream>
#include <string>
#include <vector>

using namespace std;


void getLetterFreq(string s){//this function checks the frequency of all the letters in the alphabet in a string
    int count;
    for(char q ='a';q <= 'z';q++){//this will loop through the entire alphabet
        count = 0;
        for(int i = 0; i < s.length(); i++){//compares the current letter of the alphabet to the every character in the string
        	if(s[i] < 91)//checks to see if its an upper case charecter and down cases it if it is
				s[i]+= 32;
            if(s[i] == q)//compares to the alphabet letter
                count++;
        }
        if(count!= 0){//this was added for less screen clutter so the letters with 0 frequency dont appear
			cout<<"there are "<<count<<" "<<q<<" in this string."<<endl;//prints the frequency of each letter
   		}
    }
}

void StrToTokens(string s){
	string token("");

	int lastToken = 0;
	vector<string> Token;
	for(int i = 0; i <= s.length(); i++){
		if(s[i] == ' ' || i == s.length()){//checks for delimeters or end of string
			token = s.substr(lastToken, i-lastToken);//makes a token in themo that is from the last delimiter to the current delimiter
			Token.push_back(token);//puts the tokens into a vector
			token = "";//resets the token variable
			lastToken = i+1;//sets the start of token at the next letter
		}
		
	}
	for(int i = 0; i<Token.size(); i++){//goes through the vector of tokens and prints them all
		cout<<Token[i]<<endl;
	}
	
}

int main() {
	string s("This is a very short sentence.");
	getLetterFreq(s);
	StrToTokens(s);
}
