void StrToTokens(string s) {
	//Translate String into individual tokens
	//Read S Seperating individual words by identifying white spaces
	//Check if leading character is white space
	//if (s[0] == " ")
	//Use length of string as loop delimeter
	
	int tokenCount = 0;
	char delimeter = ' ';
	string tokens[100];
	int x = 0;
	int wordLength = 0;
	 
	
	for (int i = 0; i < (s.length() + 1); i++) {
		if (s[i] == ' ' || s[i] == '\0') {
			tokens[tokenCount] = s.substr(x, wordLength);
			tokenCount++;
			x = i + 1;
			wordLength = 0;
		}
		else {
			wordLength++;
		}	 
	}


	//Print
	cout << "The output of StrToTokens(s): will be:";
	for (int i = 0; i < (tokenCount); i++) {
		cout<<", '"<< tokens[i]<<"'";
	}
