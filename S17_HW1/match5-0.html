#include <iostream>
#include <string>
using namespace std;

void StrToTokens(string);
void getLetterFreq(string);

int main() {
    string paragraph;
    cout << "Please Enter Paragraph:";
    //Collecting the input from user by calling getLine
    getline(cin, paragraph);
    cout << paragraph << "\n";
    //Determine if the input is within the given limit
    if (paragraph.length() > 100) {
        cout << "The string is limited to 100 characters. Please try again" << endl;
    } // end if statement
    getLetterFreq(paragraph);
    StrToTokens(paragraph);
}

//Function to determine which letter occurs the most frequent
void getLetterFreq(string input){
    //arrays to hold total amount in english alphabet
    int frequency [26];
    char alphabet [26] = {'a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z'};
    //Loop determine the letters input by user
    for (int i = 0; i < 26; i++) {
        int count = 0;
        //Loop to determine which letter are being used
        for (int j = 0; j < input.length(); j++) {
            if(input[j] == alphabet[i]){
                count++;
            }
        }
        //assign the frequency each letter is occuring
        frequency[i] = count;
    }
    //Loop to print out the letter and how frequent it occurs
    for(int k = 0; k < 26; k++){
        cout << "Letter " << alphabet[k] << " has frequency :" << frequency[k] << endl;
    }
}

//Function to change string into tokens
void StrToTokens(string input)
{
    cout << "\n 'Word Tokens' \n";
    string userString = " ";
    for (int i = 0; i < input.size(); i++)
    {
        // print the word if space is found and initiate next word
        if(input[i] == ' ')
        {
            cout << userString << endl;
            userString = "";
        }
        // else add to the current word
        else
        {
            userString = userString + input[i];
        }
    }
    
    cout << endl;
}
