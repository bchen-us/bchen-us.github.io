void strToTokens(string s) {
	vector<string> vect1;

	int count = 0;
	for (size_t i = 0; i < s.length() + 1; ++i) {
		if (s[i] == ' ' || s[i] == '\0') {
			string word1;

			for (size_t j = count; j < i; ++j) {
				word1 = word1 + s[j];
			}

			vect1.push_back(word1);
			count = i + 1;
		}

	}

	vector<string>::const_iterator viter = vect1.cbegin();
	for (viter; viter != vect1.cend(); ++viter)
		cout << "'" << *viter << "', ";

	cout << endl;
}
