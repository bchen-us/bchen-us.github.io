//identify and print all the tokens contained in s on the standard output 
void StrToTokens(string s){
	string list = "";
	std::vector<string> a;
	

	for (int input=0; input<s.size(); input++)
	{
		if (!isspace(s[input]))
		{
			list = list + s[input];
		}
        if (isspace(s[input]))
        {
        	if(list!=""){
        		a.push_back(list);
        	}
        	
        	list = "";
        }
	}
	a.push_back(list);
	for(int ending=0; ending<a.size(); ++ending)
		{std::cout << a[ending] +", ";}
	cout <<"\n";

}

// identify the frequency of each unique letter 
void getLetterFreq(string s)

 {
	// connvert all characters to lower case 
	transform(s.begin(), s.end(), s.begin(), ::tolower);
	// remove spaces between tokens
    s.erase(std::remove(s.begin(), s.end(), ' '), s.end());
	s.erase(std::remove(s.begin(), s.end(), '.'), s.end());
    map<char,int> chars;
    for ( int input=0; input<s.size(); input++)
        chars[s[input]]++;
    for (map<char, int>::iterator cur = chars.begin(); cur != chars.end(); ++cur)
        cout << cur->first << ": " << cur->second << ", ";
    cout <<"\n";
}
